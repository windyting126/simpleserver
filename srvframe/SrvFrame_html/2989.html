15     int CRecvAndSendMsgThread::Process(CMsg * pMsg)
16     {
17             TMsg *pTMsg = pMsg->GetTMsg();
18             WRITE_TRACE("The recvsend thread recv a  msg , msgtype %d...", pTMsg->msgType);        
19             
20             if (pTMsg->srcEntityType== SERVICEPROENTITY)
21             {
22                     if (pTMsg->msgType == SENDMSGTYPE)
23                     {
24                             CSendMsgBody *pBody = (CSendMsgBody*)pMsg->GetMsgBody();
25                             CSendTask *pSendTask = (CSendTask*)CMyserver::GetInstance()->GetClassFactory()->GenSocketTask(this,
26                                     pBody->GetTcpSocket(), SENDTASK);
27                             if (pSendTask != NULL)
28                             {
29                             	    pSendTask->Init();
30                                     CSendMsg *pSendMsg = new CSendMsg(pBody->GetSendLen(), pBody->GetSendBuf());                                
31                                     pSendTask->PutSendMsg(pSendMsg);
32                                     pSendTask->PutEndMsg();
33                                     
34                                     pBody->SetTcpSocket(NULL); //不删除通信句柄
35                                     pBody->SetSendBuf(NULL);  //不删除缓冲区
36                             }
37                     }
38                     else
39                     {
40                             WriteRunInfo::WriteLog("Send msg thread recv invalid msgtype %d", pTMsg->msgType);
41                     }
42             }
43             else if (pTMsg->srcEntityType == LISTENENTITY) //收到接收请求
44             {
45                     if (pTMsg->msgType == RECVMSGTYPE)
46                     {
47                             CRecvMsgBody *pBody = (CRecvMsgBody*)pMsg->GetMsgBody();
48                             CTcpSocket *pSocket = new CTcpSocket(pBody->GetSockAddr(), sizeof(sockaddr_in), pBody->GetSocket());
49                             CSocketTask *tmpSocketTask = NULL;
50                             tmpSocketTask = CMyserver::GetInstance()->GetClassFactory()->GenSocketTask(this,  
51                                     pSocket, RECVTASK);               
52                             tmpSocketTask->Init();         
53                             tmpSocketTask->SetTimeStat(pBody->GetTimeStat());
54                     }
55                     else
56                     {
57                             WriteRunInfo::WriteLog("Send&recv  thread recv invalid msgtype %d", pTMsg->msgType);
58                     }
59             }        
60             else if (pTMsg->srcEntityType == TIMERENTITY)
61             {
62                     TimerMsgProcess(pMsg);
63             }    
64             else
65             {
66                     WriteRunInfo::WriteLog("Send msg thread recv invalid srcEntityType %d", pTMsg->srcEntityType);
67             }
68             return 0;
69     }
