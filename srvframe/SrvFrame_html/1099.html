   22 class CSynWorkThread:public CWorkThread
   23 {
   24 public:
   25        CSynWorkThread(int iEntityType,  int iEntityId):CWorkThread(iEntityType, iEntityId)
   26        {
   27        }
   28        
   29     ~CSynWorkThread()
   30        {
   31        }
   32     
   33        int Run()
   34        {
   35              while (m_nRunFlag)
   36              {
   37                     CMsg *pMsg = m_inputList.get();
   38                     if (pMsg != NULL)
   39                     {
   40                             Process(pMsg);
   41                             delete pMsg;
   42                     }
   43              }
   44              return 0;
   45        }
   46        
   47        int GetMsgQueueLen()
   48        {
   49              return m_inputList.size();
   50        }
   51        int Process(CMsg *pMsg) = 0;
   52        int PutMsgToInputList(CMsg * pMsg)
   53        {
   54                m_inputList.put(pMsg);
   55                return 0;
   56        }
   57 protected:
   58        CQueue  m_inputList;
   59 };
