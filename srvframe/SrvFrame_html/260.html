  456         TRequestHead reqHead;
  457         int nResult;
  458         int Encode(CBufMgr &bufMgr)

  463                 return ret;
  464         }
  465         
  466         int Decode(CBufMgr &bufMgr)

  468                 int ret = 0;

  474         {
  475                 int nLen = reqHead.Size() + 4;

  477         }

  481                 reqHead.Clone(respHead.reqHead);

  574 class CMsg
  575 {
  576 public:  
  577         CMsg(TMsg &tMsg, CMsgBody *pMsgBody)
  578         {
  579                 memcpy(&m_tMsg, &tMsg, sizeof(TMsg));
  580                 m_pMsgBody = pMsgBody;
  581         }
  582         
  583         CMsg(CMsg &msg)
  584         {
  585                 memcpy(&m_tMsg, &msg.m_tMsg, sizeof(TMsg));
  586                 if (msg.m_pMsgBody != NULL)
  587                 {
  588                         m_pMsgBody = (msg.m_pMsgBody)->Clone();
  589                 }
  590                 else
  591                 {
  592                         m_pMsgBody = NULL;
  593                 }
  594         }
  595         
  596         ~CMsg()
  597         {
  598                 if (m_pMsgBody != NULL)
  599                 {
  600                         delete m_pMsgBody;
  601                 }
  602         };
  603         
  604         int Size()
  605         {
  606                 int len = sizeof(TMsg);
  607                 if (m_pMsgBody != NULL)
  608                 {
  609                         len += m_pMsgBody->Size();
  610                 }
  611                 return  len;
  612         };
  613 
  614         CMsgBody *GetMsgBody()
  615         {
  616                 return m_pMsgBody;
  617         }
  618 
  619         TMsg *GetTMsg()
  620         {
  621                 return &m_tMsg;
  622         }
