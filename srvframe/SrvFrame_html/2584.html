18     ULONG64 CSocketTask::GenTaskId()
19     {
20     	ULONG64 tmpTaskId;
21     	m_sMutex.Lock();
22     	tmpTaskId = m_sCurTaskId++;
23     	m_sMutex.UnLock();
24     	return tmpTaskId;

28     CSocketTask::CSocketTask(CTcpSocket *pTcpSocket, CNetProcessThread* pWorkThread,int nListType)
29     {                
30             m_nTaskTimeOut = CMyserver::GetInstance()->GetBaseConf().GetConfigItem().nSocketTimeout;
31             m_pTcpSocket = pTcpSocket;   
32             if (m_pTcpSocket != NULL)
33                 m_pTcpSocket->SetNonBlocking();
34             
35             m_nStartTime=CMyserver::GetInstance()->GetTimeVal().tv_sec;
36             //加入链表
37             m_lTaskId = GenTaskId();
38     
39     	//by:fwx
40     	m_nListType = nListType;		
41             m_status=0;
42             m_pThread = pWorkThread;
43     	m_pollPos=-1;
44             m_pPos = pWorkThread->AddSocketToList(this);       
45     }

47     CSocketTask::~CSocketTask()
48     {            
49             //删除所有的epoll事件
50             DelFromEpoll();
51             if (m_pTcpSocket != NULL)
52                     delete m_pTcpSocket;
53             m_pThread->DelSocketTask(this);
54     
55     		if (m_pPos != NULL)
56     			delete m_pPos;
57     };

59     int CSocketTask::CheckTimeOut()
60     {        
61             if (m_nTaskTimeOut == -1)
62             {
63                     return 0;
64             }
65             
66             int ret = 0;
67             if ((CMyserver::GetInstance()->GetTimeVal().tv_sec - m_nStartTime) > m_nTaskTimeOut)
68             {
69                     CMyserver::GetInstance()->GetCommonStat().AddNormalStat(SOCKETTIMEOUTSTAT);
70                     DestroySelf();
71                     ret = 1;
72             }
73             return ret;
74     }

77     int CSocketTask::DelFromEpoll()
78     {
79             if (m_pTcpSocket == NULL)
80             {
81                     return 0;
82             }
83             
84             CNetProcessThread *pNetThread = (CNetProcessThread*)m_pThread;
85             CEpoll *pEpoll  = pNetThread->GetEpoll();
86             pEpoll->DelFromEpoll(this);
87             return 0;
88     }
