   44 int CListenTask::HandleProcess(int events)
   45 {
   46 //        #ifndef __USE_EPOLL
   47 //        DelFromEpoll();
   48 //        #endif
   49         CListenThread *pThread = (CListenThread*)m_pThread;
   50         pThread->GetListenBlock()->mutex.Lock();
   51         if (events&(MYPOLLIN|MYPOLLRDNORM))
   52         {
   53                 struct sockaddr_in tmpAddr;
   54                 memset(&tmpAddr, 0, sizeof(sockaddr_in));
   55                 int iSocketSize=sizeof(sockaddr_in);
   56                 //int ret  = 1;
   57                 while (1)
   58                 {                                             
   59                         int iSocket = accept(m_pTcpSocket->GetSocket(), (struct sockaddr *)&tmpAddr, (socklen_t*)&iSocketSize);
   60                         if (iSocket > 0)
   61                         {
   62                                 WRITE_TRACE("Recv a client connect request");
   63                                 CMyserver::GetInstance()->GetCommonStat().AddNormalStat(RECVPKGNUMSTAT);
   64                                  
   65                                 if (CTcpSocket::GetNum() > CMyserver::GetInstance()->GetBaseConf().GetConfigItem().nMaxFdSize) //如果太忙了， 不再接入
   66                                 {
   67                                         WriteRunInfo::WriteInfoRun("The fdsize is %d, greater than %d, close the connect", CTcpSocket::GetNum(),
   68                                                 CMyserver::GetInstance()->GetBaseConf().GetConfigItem().nMaxFdSize);
   69                                         close(iSocket);
   70                                         continue;
   71                                 }
   72                                 CTimeStat timestat;
   73                                 timestat.Begin(); //打印开始的时间
   74                                 if (CMyserver::GetInstance()->GetBaseConf().GetConfigItem().nRecvTaskType == 0) //发到收发线程中去
   75                                 {
   76                                         CRecvMsgBody *pBody = new CRecvMsgBody(iSocket, tmpAddr);
   77                                         pBody->SetTimeStat(timestat);
   78                                         m_pThread->SendMsg(RECVANDSENDMSGENTITY, RANDOMENTITYID, 0, RECVMSGTYPE, pBody, 0);               
   79                                 }
   80                                 else  //发到本线程
   81                                 {
   82                                         CTcpSocket *pSocket = new CTcpSocket(tmpAddr, iSocketSize, iSocket);
   83                                         //生成一个接收数据的任务                          
   84                                         CSocketTask *pRecvTask = NULL;
   85                                         pRecvTask =  CMyserver::GetInstance()->GetClassFactory()->GenSocketTask((CNetProcessThread*)m_pThread,  
   86                                                 pSocket, RECVTASK);
   87                                                                                 if (pRecvTask != NULL)
   88                                                                                 {
   89                                                 pRecvTask->Init();
   90                                                 pRecvTask->SetTimeStat(timestat);
   91                                                                                 }
   92                                                                                 else
   93                                                                                 {
   94                                                                                         close(iSocket);
   95                                                                                 }
   96                                 }
   97                         }
   98                         else  //句柄accept完毕
   99                         {
  100                                 break;
  101                         }
  102                 }
  103         }
  104         else
  105         {
  106                 WriteRunInfo::WriteRunLog("Receive invalid event %d", events);
  107         }
  108         pThread->GetListenBlock()->mutex.UnLock();
  109 //        #ifndef __USE_EPOLL
  110 //        AddToEpoll();
  111 //        #endif
  112         return 0;
  113 }
