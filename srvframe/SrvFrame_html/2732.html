20     CLongConnectedListenThread::CLongConnectedListenThread( int iEntityType, int iEntityId, string sHost, int nPort, PLISTENBLOCK pListenBlock )
21     	: CNetProcessThread( iEntityType, iEntityId )
22     {
23     	m_sHost = sHost;
24     	m_nPort = nPort;
25     	m_pListenBlock = pListenBlock;
26     }
27     
28     CLongConnectedListenThread::~CLongConnectedListenThread()
29     {
30     
31     }

33     int CLongConnectedListenThread::Init()
34     {
35             int retValue = 0;
36             retValue = CNetProcessThread::Init();
37             if (retValue != 0)
38             {
39                     return -1;
40             }
41             m_pListenBlock->mutex.Lock();
42             if (m_pListenBlock->nSock == -1)
43             {
44                 //生成一个监听句柄
45                 int nSocket;                           
46                 retValue = CMySocket::CreateSock(SOCK_STREAM, m_sHost, m_nPort, nSocket);
47                 if (retValue < 0)
48                 {
49                         WriteRunInfo::WriteLog("Create listen sock fail %d", retValue);
50                         m_pListenBlock->mutex.UnLock();
51                         return -1;
52                 }
53                 m_pListenBlock->nSock=nSocket;
54             }
55             m_pListenBlock->mutex.UnLock();
56             
57             m_pTcpSocket = new CTcpSocket(m_sHost, m_nPort, m_pListenBlock->nSock);
58             if (m_pTcpSocket == NULL)
59             {
60                     WriteRunInfo::WriteLog("New listen socket object fail");
61                     return -1;
62             }
63             
64             //起监听端口                    
65             m_pListenTask = (CLongConnectedListenTask*)CMyserver::GetInstance()->GetClassFactory()->GenSocketTask( this, m_pTcpSocket,  LONGCONNECTEDLISTENTASK );        
66             m_pListenTask->Init();
67             if (m_pListenTask == NULL)
68             {
69                     WriteRunInfo::WriteLog("Malloc memory fail or listen fail");
70                     return -1;
71             }                      
72             return 0;
73     }
