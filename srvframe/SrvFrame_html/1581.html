  149 struct TFrameLongConnectHead
  150 {
  151         int m_nVersion;
  152         int m_nSequence;
  153         short m_nCmdId;
  154         int m_nSrcEntityType;
  155         int m_nSrcEntityId;
  156         long long m_nSrcTaskId;
  157         int m_nDstEntityType;
  158         int m_nDstEntityId;
  159         long long m_nDstTaskId;
  160         int m_nBodyLen;
  161         char szMsgBody[0];
  162 
  163         enum EFrameLongConnectHead
  164         {
  165                 FLCH_SEQOFFSET = 4,
  166                 FLCH_CMDOFFSET = 8,
  167                 FLCH_BODYLENOFFSET = 42,
  168                 FLCH_HEADLEN = 46
  169         };
  170         TFrameLongConnectHead()
  171         {
  172                 Clear();
  173         }
  174         void Clear()
  175         {
  176                 memset(this, 0, sizeof(TFrameLongConnectHead));
  177         }
  178         int GetSize()
  179         {
  180                 return FLCH_HEADLEN;
  181         }
  182         static short GetCmdId(char* pBuf, int nBufLen)
  183         {
  184                 if (nBufLen < FLCH_HEADLEN)
  185                 {
  186                         return -1;
  187                 }
  188                 return ntohs(*((short*)(pBuf + FLCH_CMDOFFSET)));
  189         }
  190         static int GetSequence(char* pBuf, int nBufLen)
  191         {
  192                 if (nBufLen < FLCH_HEADLEN)
  193                 {
  194                         return -1;
  195                 }
  196                 return  ntohl(*((int*)(pBuf + FLCH_SEQOFFSET)));
  197         }
  198         static int GetBodyLen(char* pBuf, int nBufLen)
  199         {
  200                 if (nBufLen < FLCH_HEADLEN)
  201                 {
  202                         return -1;
  203                 }
  204                 return  ntohl(*((int*)(pBuf + FLCH_BODYLENOFFSET)));
  205         }
  206         int Encode(CBufMgr& BufMgr)
  207         {
  208                 int nRet = 0;
  209                 nRet += BufMgr.PutInt(m_nVersion);
  210                 nRet += BufMgr.PutInt(m_nSequence);
  211                 nRet += BufMgr.PutShort(m_nCmdId);
  212                 nRet += BufMgr.PutInt(m_nSrcEntityType);
  213                 nRet += BufMgr.PutInt(m_nSrcEntityId);
  214                 nRet += BufMgr.PutLong64(m_nSrcTaskId);
  215                 nRet += BufMgr.PutInt(m_nDstEntityType);
  216                 nRet += BufMgr.PutInt(m_nDstEntityId);
  217                 nRet += BufMgr.PutLong64(m_nDstTaskId);
  218                 nRet += BufMgr.PutInt(m_nBodyLen);
  219                 return nRet;
  220         }       
  221         int Decode(CBufMgr& BufMgr)
  222         {
  223                 int nRet = 0;
  224                 nRet += BufMgr.GetInt(m_nVersion);
  225                 nRet += BufMgr.GetInt(m_nSequence);
  226                 nRet += BufMgr.GetShort(m_nCmdId);
  227                 nRet += BufMgr.GetInt(m_nSrcEntityType);
  228                 nRet += BufMgr.GetInt(m_nSrcEntityId);
  229                 nRet += BufMgr.GetLong64(m_nSrcTaskId);
  230                 nRet += BufMgr.GetInt(m_nDstEntityType);
  231                 nRet += BufMgr.GetInt(m_nDstEntityId);
  232                 nRet += BufMgr.GetLong64(m_nDstTaskId);
  233                 nRet += BufMgr.GetInt(m_nBodyLen);
  234                 return nRet;
  235         }
  236 };
