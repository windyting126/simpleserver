   14 int CTimerThread::Run()
   15 {
   16         while (m_nRunFlag)
   17         {
   18                 gettimeofday(&m_timeval, NULL);
   19                 m_iCurrentTimePoint = m_timeval.tv_sec%TIMERCYCLE;
   20                 m_timer.SetCurTimePoint(m_timeval.tv_sec);
   21                 CMyserver::GetInstance()->SetTimerPoint(m_iCurrentTimePoint, m_timeval);
   22                 m_timer.TimerCheck();
   23                 CMsg *pMsg = m_inputList.get();
   24                 if (pMsg != NULL)
   25                 {
   26                         Process(pMsg);
   27                         delete pMsg;
   28                 }
   29                 else
   30                 {
   31                         //定时线程实时性要求不高, 可以睡眠10ms
   32                         usleep(10000);
   33                 }
   34         }
   35         return 0;
   36 }

   49 int CTimerThread::Process(CMsg *pMsg)
   50 {
   51         TMsg *pTMsg = pMsg->GetTMsg();
   52         switch (pTMsg->msgType)
   53         {
   54                 case TIMERADDTYPE: //增加接口
   55                 {                              
   56                         CBufMsgBody *pBufMsg = (CBufMsgBody*)pMsg->GetMsgBody();
   57                         TTimeEvent *pTEvent = (TTimeEvent*)pBufMsg->GetBuf();
   58 
   59                         TTimerStr tmpTimerStr;
   60                         tmpTimerStr.iEntityId = pTMsg->srcEntityId;
   61                         tmpTimerStr.iEntityType = pTMsg->srcEntityType;
   62                         tmpTimerStr.iTaskId = pTMsg->taskId;
   63                         memcpy(&(tmpTimerStr.event), pTEvent, sizeof(TTimeEvent));                
   64                         m_timer.AddOneTimer(tmpTimerStr);
   65                         break;
   66                 }
   67                 case TIMERDELTYPE: //删除接口
   68                 {
   69                         CBufMsgBody *pBufMsg = (CBufMsgBody*)pMsg->GetMsgBody();
   70                         TTimeEvent *pTEvent = (TTimeEvent*)pBufMsg->GetBuf();
   71                         m_timer.DeleteOneTimer(pTEvent->m_nTimerPoint, pTEvent->m_iTimerId);// int iTimePoint, int iTimerId);
   72                         break;
   73                 }
   74                 case PRINTTIMER:
   75                 {
   76                         m_timer.PrintTimer();
   77                         break;
   78                 }
   79                 default:
   80                 {
   81                         WriteRunInfo::WriteLog("Timer thread , recv invaid msg %d", pTMsg->msgType);
   82                         break;
   83                 }
   84         }
   85         return 0;
   86 }
