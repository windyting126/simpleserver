17     class CAsynQueue
18     {
19     public:
20         CAsynQueue()
21         {
22             pthread_mutex_init(&m_mutex_write, NULL);
23         }
24         ~CAsynQueue()
25             {
26                     pthread_mutex_destroy(&m_mutex_write);
27             }
28     
29         int  put(CMsg *pMsg)
30          {       
31            pthread_mutex_lock(&m_mutex_write);
32            m_queue.push_back(pMsg);
33            pthread_mutex_unlock(&m_mutex_write);
34            return 0;
35        }
36         
37         CMsg*    get()
38         {   
39                 CMsg *pMsg = NULL;
40                 pthread_mutex_lock(&m_mutex_write);
41                 deque<CMsg*>::iterator tmpItr = m_queue.begin();
42                 if (tmpItr != m_queue.end())
43                 {
44                       pMsg  = *tmpItr;
45                       m_queue.pop_front();
46                 }
47                 pthread_mutex_unlock(&m_mutex_write);
48                 return pMsg;
49        }
50     
51         size_t  size()
52         {
53             size_t nSize = 0;
54             pthread_mutex_lock(&m_mutex_write);
55             nSize = m_queue.size();
56             pthread_mutex_unlock(&m_mutex_write);
57             return nSize;
58         }
59     
60     private:
61         pthread_mutex_t m_mutex_write;
62         deque<CMsg*> m_queue;               /* 保存数据的队列 */
63     };
