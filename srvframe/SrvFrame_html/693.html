32     int CRealEpoll::Init(int epollSize, int epollWaitTime)
33     {
34             m_nEpollWaitTime = epollWaitTime;
35             m_nEpollSize = epollSize;
36             m_nEpollFd = epoll_create(m_nEpollSize);
37             if (m_nEpollFd == -1)
38             {       
39                     WriteRunInfo::WriteLog("Epoll init failed, errno is %d, and error is %s", errno, strerror(errno));
40                     return -1;
41             }
42             m_epollEvents = new epoll_event[m_nEpollSize];        
43             return 0;
44     }

46     int CRealEpoll::AddToEpoll(CSocketTask *pTask, int event)
47     {
48             int tmpOprate = EPOLL_CTL_ADD;
49             struct epoll_event tmpEvent;
50             memset(&tmpEvent, 0, sizeof(epoll_event));
51             tmpEvent.events = event;
52             tmpEvent.data.ptr = pTask;
53             epoll_ctl(m_nEpollFd, tmpOprate, pTask->GetSocket(), &tmpEvent);
54             CMyserver::GetInstance()->GetCommonStat().AddNormalStat(ADDEPOLL);
55             return 0;
56     }

80     int CRealEpoll::EpollWait()
81     {
82             int  nfds =  epoll_wait(m_nEpollFd, m_epollEvents, m_nEpollSize,  m_nEpollWaitTime);
83             if (nfds == -1)
84             {
85                     WriteRunInfo::WriteRunLog("epoll_wait error(%d) is %s", errno, strerror(errno));      
86                     return 0;
87             }      
88             int tmpProcessNum=0;
89             for (int i =0; i < nfds; i++)
90             {
91                     //epoll_event *tmpEvents = m_epoll.GetEpollEvents();
92                     CSocketTask * tmpTask = (CSocketTask*)(m_epollEvents[i].data.ptr);
93                     if (tmpTask != NULL)
94                     {
95                             tmpTask->HandleProcess(m_epollEvents[i].events);
96                     }
97             }
98             tmpProcessNum = nfds;
99             return nfds;
100     }
