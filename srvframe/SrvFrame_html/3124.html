   19 CTask *CWorkThread::FindTask(ULONG64 iTaskId)
   20 {
   21         CTask *pTmpTask = NULL;
   22         map<ULONG64, CTask*>::iterator itr = m_taskList.find(iTaskId);
   23         if (itr != m_taskList.end())
   24         {
   25                 pTmpTask = itr->second;
   26         }
   27         return pTmpTask;        
   28 }

   42 int CWorkThread::TimerMsgProcess(CMsg *pMsg)
   43 {
   44         TMsg *pTMsg = pMsg->GetTMsg();
   45         CTask *pTask = FindTask(pTMsg->taskId);
   46         if (pTask != NULL)
   47         {
   48                 CBufMsgBody *pBufMsg = (CBufMsgBody*)pMsg->GetMsgBody();
   49                 TTimeEvent *pTimerEv = (TTimeEvent*)pBufMsg->GetBuf();
   50                 pTask->OnTimer(*pTimerEv);
   51         }
   52         else
   53         {
   54                 WriteRunInfo::WriteLog("The task(%llu) is not exist", pTMsg->taskId);
   55         }
   56         return 0;
   57 }

   59 void CWorkThread::SendMsg(int destEntityType, int destEntityId, ULONG64 lTaskId, int msgType,  CMsgBody *pMsgBody, ULONG64 lSrcTaskId)
   60 {
   61         TMsg tMsg;
   62         tMsg.destEntityType = destEntityType;
   63         tMsg.destEntityId = destEntityId;
   64         tMsg.srcEntityId = GetEntityId();
   65         tMsg.srcEntityType= GetEntityType();
   66         tMsg.srcTaskId = lSrcTaskId;
   67         tMsg.taskId = lTaskId;
   68         tMsg.msgType = msgType;
   69         CMsg *pMsg = new CMsg(tMsg, pMsgBody);
   70         CMyserver::GetInstance()->PutMsgToThread(pMsg);
   71 }
