31     int CShortHttpProcessThread::Process(CMsg * pMsg)
32     {
33     	TMsg *pTMsg = pMsg->GetTMsg();
34     	if ( !pTMsg )
35     	{
36     		return 0;
37     	}
38     	WriteRunInfo::WriteTrace("The ShortConnected thread recv a  msg , msgtype %d...", pTMsg->msgType);
39     
40     	if (pTMsg->srcEntityType == PROCESSENTITY)	//process message from thread with entitytype of PROCESSENTITY(wangtiefeng : 09-07-16) 
41     	{
42     		if (pTMsg->msgType == CLIENTMSGTYPE)	//process message with type of CLIENTMSGTYPE
43     		{
44     			CClientMsgBody *pBody = (CClientMsgBody *)pMsg->GetMsgBody();
45     			if ( !pBody )
46     			{
47     				return 0;
48     			}
49     			CSocketTask *pSocketTask = FindSocketTask(pTMsg->taskId);
50     			if (pSocketTask != NULL)
51     			{
52     				pSocketTask->PutMsgToSendList(pBody->GetBuffer(), pBody->GetLen());
53     				pBody->DisAttachBuf();
54     			}
55     		}
56     		else
57     		{
58     			WriteRunInfo::WriteLog("Send&recv  thread recv invalid msgtype %d", pTMsg->msgType);
59     		}
60     	}
61     	else if (pTMsg->srcEntityType == LISTENENTITY)
62     	{
63     		if (pTMsg->msgType == RECVMSGTYPE)
64     		{
65     			CRecvMsgBody *pBody = (CRecvMsgBody*)pMsg->GetMsgBody();
66     			CTcpSocket *pSocket = new CTcpSocket(pBody->GetSockAddr(), sizeof(sockaddr_in), pBody->GetSocket());
67     			CSocketTask *tmpSocketTask = NULL;
68     			tmpSocketTask = CMyserver::GetInstance()->GetClassFactory()->GenSocketTask(this,  
69     				pSocket, RECVANDSENDTASK);     //create task related to current thread(wangtiefeng : 09-07-16)          
70     			tmpSocketTask->Init();         
71     			tmpSocketTask->SetTimeStat(pBody->GetTimeStat());
72     		}
73     		else
74     		{
75     			WriteRunInfo::WriteLog("Send&recv  thread recv invalid msgtype %d", pTMsg->msgType);
76     		}
77     	}
78     	else
79     	{
80     		CNetProcessThread::Process( pMsg );
81     	}
82     	return 0;
83     }
