   21 int CNormalServer::PreStartServer()
   22 {
   23         int ret = 0;
   24 
   25                 //////////////////////////////////////////////////////////////////////////
   26                 //timer thread
   27         m_pTimerThreadGrp = new CThreadGroup(1, TIMERENTITY);
   28         ret = m_pTimerThreadGrp->Init();
   29         if (ret != 0)
   30         {       
   31                 WriteRunInfo::WriteLog("Init timer thread fail");
   32                 return -1;
   33         }
   34         WriteRunInfo::WriteLog("Init timer thread succeed");
   35 
   36                 //////////////////////////////////////////////////////////////////////////
   37                 //receive and send thread
   38         int nTcpThreadNum =  m_baseConf.GetConfigItem().nTcpThreadNum;
   39         m_pRecvAndSendThreadGrp = new CThreadGroup(nTcpThreadNum, RECVANDSENDMSGENTITY);
   40         ret = m_pRecvAndSendThreadGrp->Init();
   41         if (ret != 0)
   42         {       
   43                 WriteRunInfo::WriteLog("Init recvandsend thread fail");
   44                 return -1;
   45         }       
   46         WriteRunInfo::WriteLog("Init recvandsend thread succeed");
   47 
   48                 //////////////////////////////////////////////////////////////////////////
   49                 //maintain thread
   50         m_pMaintainThreadGrp = new CThreadGroup(1, MAINTAINENTITY);
   51         ret = m_pMaintainThreadGrp->Init();
   52         if (ret != 0)
   53         {       
   54                 WriteRunInfo::WriteLog("Init maintain thread fail");
   55                 return -1;
   56         }       
   57         WriteRunInfo::WriteLog("Init maintain thread succeed");
   58         
   59       
   60                 //////////////////////////////////////////////////////////////////////////
   61                 //监听线程
   62         int nPort = m_baseConf.GetConfigItem().nTcpPort;
   63         string sHost = m_baseConf.GetConfigItem().sServerAddr;
   64         string sTmpData = string((char*)(&nPort), 4);
   65         sTmpData.append(sHost);
   66         
   67         m_pListenThreadGrp = new CThreadGroup(1, LISTENENTITY, sTmpData);
   68         ret = m_pListenThreadGrp->Init();
   69         if (ret != 0)
   70         {       
   71                 WriteRunInfo::WriteLog("Init listen thread fail");
   72                 return -1;
   73         }
   74         WriteRunInfo::WriteLog("Init listen thread succeed");  
   75 
   76                 //////////////////////////////////////////////////////////////////////////
   77                 //
   78         m_pMaintainUpdThread = new CMaintainUdpThread(m_baseConf.GetConfigItem().sMaintainIp, 
   79                         m_baseConf.GetConfigItem().nMaintainPort);
   80         ret = m_pMaintainUpdThread->Start();
   81         if (ret != 0)
   82         {
   83                 WriteRunInfo::WriteLog("Init maintain udp thread fail");
   84                 return -1;
   85         }        
   86         WriteRunInfo::WriteLog("Init maintain udp thread succeed");                        
   87         return 0;
   88 }
