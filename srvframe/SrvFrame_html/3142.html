  117 template <class T1, class T2=CDefaultRes> 
  118 class CBaseCommand:public CCommand
  119 {
  120         public:
  121                 static int GetCommandId(char *pBuf, int nLen) //zz
  122                 {
  123                         if(nLen < REQHEADLEN) 
  124                         {
  125                                 WriteRunInfo::WriteLog("The cmd buf len is less than %d, "
  126                                                 "too short", REQHEADLEN);
  127                                 return INVALID_CMD_ID;
  128                         }
  129 
  130                         unsigned short sCmdId;
  131                         memcpy(&sCmdId, pBuf + CMDIDOFF, sizeof(sCmdId));
  132                         return  ntohs(sCmdId);
  133                 }
  134                 CBaseCommand(char *pBuf, int nLen):m_outBuf( NULL , 0 , 0 )
  135                 {
  136                         m_outHead.m_nProtocolVersion = VOD_PROTOCOL_VER;
  137                         m_inBuf.AttachBuf(pBuf, nLen);
  138             m_nIfDecode=0;
  139             m_nResult = -1;
  140                 }
  141                 virtual ~CBaseCommand(){}
  142 
  143                 int Process2()
  144                 {
  145                         return ProcessCmd();
  146                 }
  147 
  148                 virtual int Process(int &nResult) //modified by dijie 09-11-4
  149                 {
  150                         //解码
  151                         nResult = 0;
  152                     int nRet=Decode();
  153             if (nRet == 0)
  154             {
  155                 nResult =  ProcessCmd();
  156                 m_nResult = nResult;
  157                 if (nResult == 1)
  158                     m_nResult = 0;
  159             }
  160             
  161             if (nResult <= 1 && m_nIfRes==1) //>0一般是暂时不需要进行处理
  162             {   
  163                 GenResHead(); //生成返回的头部
  164                 nRet=Encode();
  165             }                   
  166                         return nRet;
  167                 }
  168 
  169                 T1 &GetIn()
  170                 {
  171                         return m_in;
  172                 }
  173 
  174                 char *GetResBuf(int &nLen)
  175                 {
  176                         nLen = m_outBuf.GetRealBufLen();
  177                         return (char*)m_outBuf.GetBuf();
  178                 }
  179 
  180                 void DisAttachBuf()
  181                 {
  182                         m_outBuf.DisAttachBuf();
  183                 }
  184 
  185                 short GetMsgType()
  186                         {
  187                         return m_inHead.m_nCmd;
  188                         }
  189         protected:
  190         void GenResHead()
  191         {
  192             m_outHead.m_nProtocolVersion = m_inHead.m_nProtocolVersion;
  193             m_outHead.m_nSeqence = m_inHead.m_nSeqence;
  194             m_outHead.m_nBodylen = 3;
  195             if (m_nResult == 0)
  196             {
  197                 m_outHead.m_nBodylen += m_out.GetSize();
  198             }
  199         }
  200                 int Encode()
  201                 {
  202                         int iPacketLength = sizeof(SERVERINNERCOMMANDPACKET) + m_out.GetSize() + 3*sizeof(int) + sizeof(short) + sizeof( char );
  203                         char * pEncodeBuf = new char[iPacketLength];
  204                         m_outBuf.AttachBuf( pEncodeBuf , iPacketLength );
  205                         m_outBuf.SetOffset( sizeof(SERVERINNERCOMMANDPACKET) );
  206 
  207                         int ret=0;
  208                         ret+=m_outBuf.PutInt(m_outHead.m_nProtocolVersion);
  209                         ret+=m_outBuf.PutInt(m_outHead.m_nSeqence);
  210                         ret+=m_outBuf.PutInt(m_outHead.m_nBodylen);
  211                         ret+=m_outBuf.PutShort(m_outHead.m_nCmd);
  212                         ret+=m_outBuf.PutChar(m_nResult);
  213                         if (m_nResult == 0)
  214                         {                               
  215                                 ret += m_out.Encode(m_outBuf);
  216                         }
  217                         return ret;
  218                 }
  219                 int Decode()
  220                 {
  221                         int ret=0;
  222             if (m_nIfDecode == 0)
  223                 {
  224                         ret+=m_inBuf.GetInt(m_inHead.m_nProtocolVersion);
  225                         ret+=m_inBuf.GetInt(m_inHead.m_nSeqence);
  226                         ret+=m_inBuf.GetInt(m_inHead.m_nBodylen);
  227                         ret+=m_inBuf.GetShort(m_inHead.m_nCmd);
  228                         if (m_inHead.m_nBodylen + 12 != m_inBuf.GetBufLen())
  229                         {
  230                                 WriteRunInfo::WriteLog("The body len is invalid, %d != %d",
  231                                                 m_inHead.m_nBodylen + 12, m_inBuf.GetBufLen());
  232                                 return -1;
  233                         }
  234                         ret+=m_in.Decode(m_inBuf);      
  235             m_nIfDecode = 1;
  236                 }            
  237                         return ret;
  238                 }
  239 
  240 
  241                 virtual int ProcessCmd() = 0;
  242                 int m_nIfDecode;
  243                 CBaseCommandHead m_inHead;      
  244                 CBaseCommandHead m_outHead;     
  245                 CBufMgr m_inBuf;
  246                 CTmpBufMgr m_outBuf;
  247                 int m_nResult;
  248                 T1 m_in;
  249                 T2 m_out;
  250 };
