190     struct TConDstSrvInfo
191     {
192     	unsigned char m_nServerType;
193     	unsigned short m_nServerIndex;
194     	//function
195     	TConDstSrvInfo() : m_nServerType(0), m_nServerIndex(0)
196     	{
197     
198     	}
199     	TConDstSrvInfo(unsigned char nServerType, unsigned short nServerIndex)
200     	{
201     		m_nServerType = nServerType;
202     		m_nServerIndex = nServerIndex;
203     	}
204     	unsigned int GetWeight() const
205     	{
206     		unsigned int nWeight = ((m_nServerType << 16) & 0xffff0000) + m_nServerIndex;
207     		return nWeight;
208     	}
209     	bool operator<(const TConDstSrvInfo& ConDstSrvInfo) const
210     	{
211     		return (GetWeight() < ConDstSrvInfo.GetWeight());
212     	}
213     	bool operator==(const TConDstSrvInfo& ConDstSrvInfo) const
214     	{
215     		return ((m_nServerType == ConDstSrvInfo.m_nServerType) && (m_nServerIndex == ConDstSrvInfo.m_nServerIndex));
216     	}
217     };

276     class CEpollMsg
277     {
278     public:
279     	CEpollMsg()
280     	{
281     		m_nMsgType = -1;
282     		m_pMsgBody = NULL;
283     	}
284     	CEpollMsg(int nMsgType, TEpollMsgBody *pMsgBody)
285     	{
286     		m_nMsgType = nMsgType;
287     		m_pMsgBody = pMsgBody;
288     	}
289     	~CEpollMsg()
290     	{
291     		if (m_pMsgBody)
292     		{
293     			delete m_pMsgBody;
294     		}
295     	}
296     public:
297     	int GetMsgType() {return m_nMsgType;}
298     	TEpollMsgBody * GetMsgBody() {return m_pMsgBody;}
299     	void DetachMsgBody() {m_pMsgBody = NULL;}
300     private:
301     	int m_nMsgType;
302     	TEpollMsgBody *m_pMsgBody;
303     };
