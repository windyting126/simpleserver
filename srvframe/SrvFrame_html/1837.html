  162 void sighandler(int signum)
  163 {
  164         sigset_t block_mask;
  165 
  166         if (signum == SIGCHLD)
  167         {
  168                 int errno2, result; 
  169                 pid_t deadpid;
  170                 errno2 = errno;
  171                 while((deadpid = waitpid(0, &result, WNOHANG)) > 0)
  172                 {
  173                 }
  174                 errno = errno2;
  175                 signal(SIGCHLD, sighandler);
  176         }
  177         else if (signum == SIGUSR1)
  178         {
  179                 signal(SIGUSR1, sighandler);
  180         }
  181         else if ( signum == SIGINT || signum == SIGTERM)
  182         {                                                
  183                 /*屏蔽所有可能的信号*/
  184                 sigemptyset( &block_mask );
  185                 sigaddset( &block_mask, SIGINT );
  186                 sigaddset( &block_mask, SIGQUIT );
  187                 sigaddset( &block_mask, SIGTERM );
  188                 sigaddset( &block_mask, SIGCHLD );
  189                 sigaddset( &block_mask, SIGIO );
  190                 sigaddset( &block_mask, SIGPIPE );
  191                 sigaddset( &block_mask, SIGHUP );
  192                 sigaddset( &block_mask, SIGSEGV);
  193                 sigprocmask( SIG_BLOCK, &block_mask, NULL );
  194                 exit(0);
  195         }
  196 }

  198 int CMyserver::CommonStartServer()
  199 {
  200         int ret = 0;
  201         //定时线程
  202         CThreadGroup * pTimerThreadGrp = new CThreadGroup(1, TIMERENTITY);
  203         ret = pTimerThreadGrp->Init();
  204         if (ret != 0)
  205         {       
  206                 WriteRunInfo::WriteLog("Init timer thread fail");
  207                 return -1;
  208         }
  209         WriteRunInfo::WriteLog("Init timer thread succeed");
  210 
  211         //收发线程组
  212         int nTcpThreadNum =  m_baseConf.GetConfigItem().nTcpThreadNum;
  213         CThreadGroup *pRecvAndSendThreadGrp = new CThreadGroup(nTcpThreadNum, RECVANDSENDMSGENTITY);
  214         ret = pRecvAndSendThreadGrp->Init();
  215         if (ret != 0)
  216         {       
  217                 WriteRunInfo::WriteLog("Init recvandsend thread fail");
  218                 return -1;
  219         }       
  220         WriteRunInfo::WriteLog("Init recvandsend thread succeed");
  221 
  222         //维护线程
  223         CThreadGroup * pMaintainThreadGrp = new CThreadGroup(1, MAINTAINENTITY);
  224         ret = pMaintainThreadGrp->Init();
  225         if (ret != 0)
  226         {       
  227                 WriteRunInfo::WriteLog("Init maintain thread fail");
  228                 return -1;
  229         }       
  230         WriteRunInfo::WriteLog("Init maintain thread succeed");
  231 
  232         //监听线程
  233         int nPort = m_baseConf.GetConfigItem().nTcpPort;
  234         string sHost = m_baseConf.GetConfigItem().sServerAddr;
  235         string sTmpData = string((char*)(&nPort), 4);
  236         sTmpData.append(sHost);
  237 
  238         CThreadGroup *pListenThreadGrp = new CThreadGroup(1, LISTENENTITY, sTmpData);
  239         ret = pListenThreadGrp->Init();
  240         if (ret != 0)
  241         {       
  242                 WriteRunInfo::WriteLog("Init listen thread fail");
  243                 return -1;
  244         }
  245         WriteRunInfo::WriteLog("Init listen thread succeed");                
  246         return 0;
  247 }

  249 void CMyserver::WriteMsgQueueLog(int nLogType)
  250 {
  251         stringstream statStr;
  252         statStr.str("");
  253         statStr << "<<server msgqueue len>>:\n";
  254         for (int nA = 0; nA < MAXENTITYNUM; nA ++)
  255         {
  256                 if (m_threadGroupList[nA] == NULL)
  257                 {
  258                         continue;
  259                 }
  260                 statStr << "\t<<EntityType:" <<  nA << ">>:";
  261                 vector<CWorkThread *>* pVec = m_threadGroupList[nA]->GetThreadList();
  262                 int nThreadNum = (int)pVec->size();
  263                 for (int nB = 0; nB < nThreadNum; nB ++)
  264                 {
  265                         statStr << "[" << nB << ":" << (*pVec)[nB]->GetMsgQueueLen() << "]";
  266                 }
  267                 statStr << "\n";
  268         }
  269         WRITE_ALL_LOG(nLogType, "%s", statStr.str().c_str());
  270 }
