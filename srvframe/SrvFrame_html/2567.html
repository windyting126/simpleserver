31     int CLongConnListenTask::HandleProcess(int events)
32     {
33     	if (events & (MYPOLLIN | MYPOLLRDNORM))
34     	{
35     		struct sockaddr_in tmpAddr;
36     		memset(&tmpAddr, 0, sizeof(sockaddr_in));
37     		int iSocketSize = sizeof(sockaddr_in);
38     		while (1)
39     		{                                 
40     			int iSocket = accept(m_pTcpSocket->GetSocket(), (struct sockaddr *)&tmpAddr, (socklen_t*)&iSocketSize);
41     			if (iSocket > 0)
42     			{
43     				WRITE_TRACE("Longconnect Recv a client connect request");
44     				CMyserver::GetInstance()->GetCommonStat().AddNormalStat(RECVPKGNUMSTAT);
45     				 
46     				CTimeStat timestat;
47     				timestat.Begin(); //打印开始的时间
48     
49     				CRecvMsgBody *pBody = new CRecvMsgBody(iSocket, tmpAddr);
50     				pBody->SetTimeStat(timestat);
51     				m_pThread->SendMsg(m_nRecvAndSendThreadType, RANDOMENTITYID, 0, MSGTYPE_LONGCONN_ACCEPT, pBody, 0);
52     			}
53     			else  //句柄accept完毕
54     			{
55     				break;
56     			}
57     		}
58     	}
59     	else
60     	{
61     		WriteRunInfo::WriteRunLog("Receive invalid event %d", events);
62     	}
63     	return 0;
64     }
