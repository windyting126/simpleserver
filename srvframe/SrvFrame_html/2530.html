   34 int CRecvTask::HandleProcess(int events)
   35 {
   36 //        DelFromEpoll();
   37         if ((events&MYPOLLERR) == MYPOLLERR || (events&MYPOLLHUP) == MYPOLLHUP)
   38         {
   39                 CMyserver::GetInstance()->GetCommonStat().AddNormalStat(SOCKETERRORTYPE);
   40                 WriteRunInfo::WriteLog("Recv a invalid data pack, event %d, close the socket %d", events, 
   41                         m_pTcpSocket->GetSocket());
   42                 DestroySelf();
   43                 return 0;
   44         }
   45         int ret = 0;
   46         if (events& (MYPOLLIN|MYPOLLRDNORM)) //读事件
   47         {
   48                 ret = RecvProcess();
   49                 if (ret != 0)
   50                 {
   51                         if (ret == -1)                                
   52                                 WriteRunInfo::WriteLog("The socket occur error");
   53                         
   54                         DestroySelf();
   55                         return -1;
   56                 }
   57         }
   58         else //错误的事件
   59         {
   60                //写日志
   61                WriteRunInfo::WriteLog("Error epoll event, %d", events);
   62                DestroySelf();
   63                return -1;
   64         }                
   65 //        AddToEpoll();
   66         return 0;
   67 }

   74 int CRecvTask::RecvBody()
   75 {
   76         int retValue = 0;
   77         int ret = 0;
   78         //接着接收消息体
   79         int recvLen = (MAXRECVLEN > m_nRecvLen - m_iHasRecvLen)?(m_nRecvLen - m_iHasRecvLen):MAXRECVLEN;
   80         WRITE_TRACE("recvlen is %d", recvLen);
   81         ret = recv(m_pTcpSocket->GetSocket(), m_recvBuf, recvLen, MSG_DONTWAIT);
   82         if (ret == 0) //对方关闭端口
   83         {
   84                 //写日志
   85                 WriteRunInfo::WriteLog("The client has closed the socket, the socket %d is invalid\n", m_pTcpSocket->GetSocket());
   86                 return -1;
   87         }
   88         else if (ret == recvLen)  //接收成功
   89         {
   90                 m_iHasRecvLen += ret;                
   91                 m_sRecvStr.append(m_recvBuf, ret);
   92                 if (m_sRecvStr.length() >= m_nRecvLen)
   93                 {
   94                         retValue = 1; //接收完毕
   95                 }
   96                 else
   97                 {
   98                         retValue = 0; //还没有接收完毕
   99                 }
  100         }
  101         else if (ret > 0)  //部分接收成功
  102         {
  103                 m_iHasRecvLen += ret;
  104                 retValue = 0;
  105                 m_sRecvStr.append(m_recvBuf, ret);
  106         }
  107         else // < 0
  108         {
  109                 //将一些错误过滤
  110                 if (errno != EINTR && errno != EAGAIN)
  111                 {
  112                         retValue = -1;
  113                 }          
  114                 else
  115                 {
  116                         retValue = 0;
  117                 }
  118         }
  119         return retValue;
  120 }

  123 int CRecvTask::RecvHttpRequest()
  124 {       
  125         int retValue = 0;
  126         int ret = 0;
  127         //接着接收消息体
  128         int recvLen = MAXHTTPHEADLEN - m_iHasRecvLen;
  129         ret = recv(m_pTcpSocket->GetSocket(), m_recvBuf, recvLen, MSG_DONTWAIT);
  130         if (ret == 0) //对方关闭端口
  131         {
  132                 //写日志
  133                 WriteRunInfo::WriteLog("The client has closed the socket, the socket %d is invalid\n",  m_pTcpSocket->GetSocket());
  134                 return -1;
  135         }
  136         else if (ret > 0) //
  137         {
  138                 //判断下http头是否接收完毕
  139                 m_iHasRecvLen += ret;
  140                 m_sRecvStr.append(m_recvBuf, ret);                
  141                 char *tmp = (char*)strstr(m_sRecvStr.c_str(), "\r\n\r\n"); //连续两个换行
  142                 if (tmp == NULL) //http头没有结束
  143                 {
  144                         if (m_sRecvStr.length() >= MAXHTTPHEADLEN) //超过了最大http头子的长度
  145                         {
  146                                 WriteRunInfo::WriteLog("The http  head is invalid, the len is greater than maxlen %d", MAXHTTPHEADLEN);
  147                                 return -1;
  148                         }
  149                         else   //http头没有结束
  150                         {
  151                                 retValue = 0; //继续是READHTTPHEADSTATUS
  152                         }
  153                 }
  154                 else //http头结束了
  155                 {
  156                         m_sRecvHead = string(m_sRecvStr,0, (int)(tmp-m_sRecvStr.c_str()));
  157                         string sTmpLen;
  158                         int ret = CToolKit::GetStringByLabel(m_sRecvStr, "Content-Length:", "\r\n", sTmpLen);
  159                         if (ret != 0)
  160                         {
  161                                 WriteRunInfo::WriteLog("The http head is invalid, it not include Content-Length item");
  162                                 return -1;
  163                         }
  164                         unsigned int nConLen = atoi(sTmpLen.c_str());
  165                         if (nConLen > MAXRECVLEN || nConLen <= 0)
  166                         {
  167                                 WriteRunInfo::WriteLog("The http bodylen(%d) is invalid", nConLen);
  168                                 return -1;
  169                         }
  170                         int nLeftLen = tmp - m_sRecvStr.c_str() + 4 + nConLen - m_sRecvStr.length();
  171                         if (nLeftLen > 0)
  172                         {
  173                                int tmpBodyLen = m_sRecvStr.length() -(tmp - m_sRecvStr.c_str()) - 4;
  174                                char *pTmpbody = new char[tmpBodyLen];
  175                                CBufMgr bufMgr1(pTmpbody);
  176                                memcpy(pTmpbody, tmp + 4, tmpBodyLen);
  177                                m_sRecvStr = string(pTmpbody, tmpBodyLen);
  178                                m_nRecvLen= nConLen;
  179                                m_iHasRecvLen = tmpBodyLen;
  180                                retValue = 0;  //表示接收未完成标志
  181                                m_status = READMSGBODYSTATUS;
  182                         }
  183                         else  //包体已经接收完了
  184                         {                                               
  185                                char *pTmpbody = new char[nConLen];
  186                                CBufMgr bufMgr1(pTmpbody);
  187                                memcpy(pTmpbody, tmp + 4, nConLen);
  188                                m_sRecvStr = string(pTmpbody, nConLen);
  189                                m_nRecvLen= nConLen;
  190                                m_iHasRecvLen = nConLen;
  191                                retValue = 1;  //表示接收完成标志
  192                         }
  193                 }                
  194         }       
  195         else // < 0
  196         {
  197                 //继续接收消息
  198                 if (errno != EINTR && errno != EAGAIN)
  199                 {
  200                         WriteRunInfo::WriteLog("Recv fail, error(%d) is %s", errno, strerror(errno));
  201                         retValue = -1;
  202                 }          
  203                 else
  204                 {
  205                         retValue = 0;
  206                 }
  207         }                
  208         return retValue;
  209 }

  211 int CRecvTask::RecvGetHttpRequest()
  212 {
  213         int retValue = 0;
  214         int ret = 0;
  215         //接着接收消息体
  216         int recvLen = MAXHTTPHEADLEN - m_iHasRecvLen;
  217         ret = recv(m_pTcpSocket->GetSocket(), m_recvBuf, recvLen, MSG_DONTWAIT);
  218         if (ret == 0) //对方关闭端口
  219         {
  220                 //写日志
  221                 WriteRunInfo::WriteLog("The client has closed the socket, the socket %d is invalid\n",  m_pTcpSocket->GetSocket());
  222                 return -1;
  223         }
  224         else if (ret > 0) //
  225         {
  226                 m_iHasRecvLen += ret;
  227                 m_sRecvStr.append(m_recvBuf, ret); 
  228                 //unsigned int nPos = m_sRecvStr.find("\r\n\r\n");
  229                 char *pTmp = (char*)strstr(m_sRecvStr.c_str(), "\r\n\r\n");
  230                 if (pTmp != NULL) //接收完成
  231                 {
  232                         retValue = 1;
  233                 }
  234                 else
  235                 {
  236                         m_status = READGETHTTPSTATUS;
  237                 }
  238         }
  239         else
  240         {
  241                         //继续接收消息
  242                 if (errno != EINTR && errno != EAGAIN)
  243                 {
  244                         WriteRunInfo::WriteLog("Recv fail, error(%d) is %s", errno, strerror(errno));
  245                         retValue = -1;
  246                 }          
  247                 else
  248                 {
  249                         retValue = 0;
  250                 }
  251         }
  252         return retValue;
  253 }

  286 int CRecvTask::AddToEpoll()

  289         {
  290                 return 0;
  291         }
