26     int CListenThread::Init()
27     {
28             int retValue = 0;
29             retValue = CNetProcessThread::Init();
30             if (retValue != 0)
31             {
32                     return -1;
33             }
34             m_pListenBlock->mutex.Lock();
35             if (m_pListenBlock->nSock == -1)
36             {
37                 //生成一个监听句柄
38                 int nSocket;                           
39                 retValue = CMySocket::CreateSock(SOCK_STREAM, m_sHost, m_nPort, nSocket);
40                 if (retValue < 0)
41                 {
42                         WriteRunInfo::WriteLog("Create listen sock fail %d", retValue);
43                         m_pListenBlock->mutex.UnLock();
44                         return -1;
45                 }
46                 m_pListenBlock->nSock=nSocket;
47             }
48             m_pListenBlock->mutex.UnLock();
49             
50             m_pTcpSocket = new CTcpSocket(m_sHost, m_nPort, m_pListenBlock->nSock);
51             if (m_pTcpSocket == NULL)
52             {
53                     WriteRunInfo::WriteLog("New listen socket object fail");
54                     return -1;
55             }
56             
57             //起监听端口                    
58             m_pListenTask = (CListenTask*)CMyserver::GetInstance()->GetClassFactory()->GenSocketTask(this, m_pTcpSocket,  LISTENTASK);        
59             m_pListenTask->Init();
60             if (m_pListenTask == NULL)
61             {
62                     WriteRunInfo::WriteLog("Malloc memory fail or listen fail");
63                     return -1;
64             }                      
65             return 0;
66     }
