37     int CMyHttpCmd::GetParaList()
38     {
39             CToolKit::GetStringByLabel(m_sInHead, "Host:", "\r\n", m_sHostDomain);
40             CToolKit::StringTrim(m_sHostDomain);   
41             
42     
43             string sPara;
44             CToolKit::GetStringByLabel(m_sUrl, "?", "", sPara);
45             vector<string> strList;
46             string sDelim = "&";
47             CToolKit::SplitString(sPara, sDelim, strList);
48             int nTmpNum = strList.size();
49             int i=0;
50             for ( i=0; i < nTmpNum; i++)
51             {
52                     string sParaName;
53                     string sParaValue;
54                     CToolKit::GetStringByLabel(strList[i], "", "=", sParaName);
55                     CToolKit::GetStringByLabel(strList[i], "=", "", sParaValue);
56                     if (sParaName != "")
57                     {
58                             CToolKit::StringTrim(sParaName);
59                             CToolKit::StringTrim(sParaValue);
60                             m_paraList.insert(make_pair(sParaName, sParaValue));
61                     }
62             }
63     
64             
65             //获取cookie
66             strList.clear();
67             sDelim = ";";
68             string sCookieStr;
69             CToolKit::GetStringByLabel(m_sInHead, "Cookie:",  "\r\n", sCookieStr);
70             if (sCookieStr != "")
71             {
72                     CToolKit::StringTrim(sCookieStr);
73                     CToolKit::SplitString(sCookieStr, sDelim, strList);
74                     nTmpNum = strList.size();
75                     for ( i=0; i < nTmpNum; i++)
76                     {
77                             string sParaName;
78                             string sParaValue;
79                             CToolKit::GetStringByLabel(strList[i], "", "=", sParaName);
80                             CToolKit::GetStringByLabel(strList[i], "=", "", sParaValue);
81                             if (sParaName != "")
82                             {
83                                     CToolKit::StringTrim(sParaName);
84                                     CToolKit::StringTrim(sParaValue);
85                                     m_cookieList.insert(make_pair(sParaName, sParaValue));
86                             }
87                     }
88             }
89     
90             CToolKit::GetStringByLabel(m_sInHead, "Referer:", "\r\n", m_sReferUrl);
91             return 0;
92     }

95     int CMyHttpCmd::ProcessCmd()
96     {               
97             //获取参数列表
98             if (m_nIfExtra == 1)
99                 GetParaList();
100     
101             int ret  = 0;
102             ret = Decode();
103             if (ret == 0)
104             {
105                     Process();
106             }
107             
108             if (m_nIfSendRes == 1)
109             {
110                     ret = Encode();                    
111                     //生成http头             
112                     m_sOut = m_sResHeader +
113                             "Connection: close\r\n"
114                             "Accept-Ranges: bytes\r\n"
115                             "Server: MyHttpSrv(1.0)\r\n"
116                             "Date:" + CToolKit::SecToHttpTime(time(NULL)) + "\r\n"
117                             "Content-Length: " + CToolKit::IntToStr(m_sOutBody.length()) + "\r\n";
118                                    
119                     m_sOut += "\r\n";
120                     if (m_sOutBody.length() > 0)
121                             m_sOut += m_sOutBody;
122             }
123             return 0;
124     }
