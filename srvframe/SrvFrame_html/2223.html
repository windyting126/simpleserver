29     int CCommonTask::CheckConnectState( unsigned char ucMachineType , unsigned short usMachineIndex )
30     {
31     	if (usMachineIndex >= 256)
32     	{
33     		return CONNECT_ATTRIBUTE_CLOSE;
34     	}
35     	int nConnectStat = 0;
36     	m_lockTaskConnectAttrib.GetReadLock();
37     	nConnectStat = m_TaskConnectAttrib[ucMachineType][usMachineIndex].m_nConnectState;
38     	m_lockTaskConnectAttrib.ReleaseLock();
39     	return nConnectStat;
40     }

42     int CCommonTask::GetConnectTaskId(unsigned char ucMachineType , unsigned short usMachineIndex, ULONG64 &nTaskId)
43     {
44     	if (usMachineIndex >= 256)
45     	{
46     		return -1;
47     	}
48     	m_lockTaskConnectAttrib.GetReadLock();
49     	if (m_TaskConnectAttrib[ucMachineType][usMachineIndex].m_nConnectState == CONNECT_ATTRIBUTE_CLOSE)
50     	{
51     		m_lockTaskConnectAttrib.ReleaseLock();
52     		return -1;
53     	}
54     	nTaskId = m_TaskConnectAttrib[ucMachineType][usMachineIndex].m_nTaskId;
55     	m_lockTaskConnectAttrib.ReleaseLock();
56     	return 0;
57     }

60     void CCommonTask::AddTaskConnectAttrib( unsigned char ucMachineType , unsigned short usMachineIndex , int iConnectState)
61     {
62     	if (usMachineIndex >= 256 )
63     	{
64     		WriteRunInfo::WriteError("usMachineIndex=%u too large", usMachineIndex);
65     		return;
66     	}
67     
68     	m_lockTaskConnectAttrib.GetWriteLock();
69     	m_TaskConnectAttrib[ucMachineType][usMachineIndex].m_nConnectState = iConnectState;
70     	m_TaskConnectAttrib[ucMachineType][usMachineIndex].m_nTaskId = GetTaskId();
71     	m_lockTaskConnectAttrib.ReleaseLock();
72     }

74     CCommonTask::CCommonTask( CTcpSocket *pSocket, CNetProcessThread *pWorkThread, unsigned char ucMachineType, unsigned short usMachineIndex, bool bActive, int iListType )
75     	: CRecvAndSendTask( pSocket, pWorkThread, iListType )
76     	, m_uReceivePos( 0 )	
77     	, m_uReceiveSize( 0 )
78     	, m_ucDestMachineType( ucMachineType )
79     	, m_usDestMachineIndex( usMachineIndex )
80     	, m_bActive( bActive )
81     	, m_state( ReadPacketHead )
82     {
83     	ResetRecvBuffer( 0, sizeof( SERVERCOMMANDPACKET ) );
84     	AddTaskConnectAttrib( ucMachineType , usMachineIndex , CONNECT_ATTRIBUTE_ESTABLISH );
85     	m_nLastRecvPinTime = (long long)CMyserver::GetInstance()->GetTimeVal().tv_sec;
86     	m_nTaskTimeOut = CMyserver::GetInstance()->GetBaseConf().GetConfigItem().nLongConnectTimeout;  //汪铁丰：2010-12-06 增加了定时ping逻辑，修正无法判断连接断开的bug
87     	if (m_pTcpSocket != NULL)
88     	{
89     		int nSocket = m_pTcpSocket->GetSocket();
90     		int nSockBufLen = 8 * 1024 * 1024;//set socket buflen 8MB
91     		setsockopt(nSocket, SOL_SOCKET, SO_SNDBUF, &nSockBufLen, sizeof(int));
92     		setsockopt(nSocket, SOL_SOCKET, SO_RCVBUF, &nSockBufLen, sizeof(int));
93     	}
94     }

105     int CCommonTask::Init()
106     {
107     	int ikeepAlive = 1; // 开启keepalive属性
108     	int ikeepIdle = 60; // 如该连接在60秒内没有任何数据往来,则进行探测 
109     	int ikeepInterval = 5; // 探测时发包的时间间隔为5 秒
110     	int ikeepCount = 3; // 探测尝试的次数.如果第1次探测包就收到响应了,则后2次的不再发.
111     
112     	setsockopt( GetSocket(), SOL_SOCKET, SO_KEEPALIVE, ( void * )&ikeepAlive, sizeof( ikeepAlive ) );
113     	setsockopt( GetSocket(), SOL_TCP, TCP_KEEPIDLE, ( void* )&ikeepIdle, sizeof( ikeepIdle ) );
114     	setsockopt( GetSocket(), SOL_TCP, TCP_KEEPINTVL, ( void * )&ikeepInterval, sizeof( ikeepInterval ) );
115     	setsockopt( GetSocket(), SOL_TCP, TCP_KEEPCNT, ( void * )&ikeepCount, sizeof( ikeepCount ) );
116     	
117     	AddToEpoll();
118     	return 0;
119     }

127     int CCommonTask::CheckTimeOut()
128     {
129     	if (m_nTaskTimeOut == -1)
130     	{
131     		return 0;
132     	}
133     
134     	int ret = 0;
135     	if ((CMyserver::GetInstance()->GetTimeVal().tv_sec - m_nLastRecvPinTime) > m_nTaskTimeOut)
136     	{
137     		DestroySelf();
138     		ret = 1;
139     	}
140     	return ret;
141     }

143     int CCommonTask::RecvProcess( void )
144     {
145     	unsigned int uMaxReceive;
146     	int iReced;
147     
148         int ii=0;
149         while (ii < LONGCONNECT_RECVSNED_NUM)
150         {
151         	uMaxReceive = m_uReceiveSize - m_uReceivePos;
152         	iReced = recv( GetSocket(), &m_vReceiveBuffer[0] + m_uReceivePos,  uMaxReceive, MSG_DONTWAIT );
153         	if ( iReced == 0 )
154         	{
155         	    WriteRunInfo::WriteLog("client close the socket, %d", GetSocket());
156         		return -1;
157         	}
158         	
159         	if ( iReced < 0 )
160         	{
161         		if ( errno != EINTR && errno != EAGAIN )
162         		{
163         			return -1;
164         		}
165         		return 0;
166         	}
167         	
168         	m_uReceivePos += iReced;
169         	
170         	switch( m_state ) 
171         	{
172         	case ReadPacketHead:
173         		{
174         			if ( !PacketFinished() ) break;
175         			PSERVERCOMMANDPACKET pServerCommandPacket = ( PSERVERCOMMANDPACKET )&m_vReceiveBuffer[0];
176         			int iNeedRecvLen = ( int )ntohl( pServerCommandPacket->iMsgLen );
177         			if ( iNeedRecvLen > ( 400 * 1024 * 1024 ) || iNeedRecvLen < 0 )
178         			{
179         				return -1;
180         			}
181     				if (iNeedRecvLen == 0)
182     				{
183     					m_nLastRecvPinTime = CMyserver::GetInstance()->GetTimeVal().tv_sec;
184     					//printf("Update tasktime recv %lld(threadid=%d:%d,taskid=%llu)\n", m_nLastRecvPinTime, m_pThread->GetEntityType(),m_pThread->GetEntityId() ,m_lTaskId);
185     					ResetRecvBuffer( 0, sizeof( SERVERCOMMANDPACKET ) );
186     					
187     					SERVERINNERCOMMANDPACKET srvCommHeader;
188     					memset(&srvCommHeader, 0, sizeof(SERVERINNERCOMMANDPACKET));
189     					memcpy(&(srvCommHeader.serverCommandPacket), pServerCommandPacket, sizeof(SERVERCOMMANDPACKET));
190     					if (srvCommHeader.serverCommandPacket.usMsgId == (unsigned short)(-1))
191     					{
192     						srvCommHeader.serverCommandPacket.usMsgId = (unsigned short)htons (-2);	//修改ping回报消息id，避免ping包造成死循环
193     						PutMsgToSendList((char*)&srvCommHeader, (int)sizeof(srvCommHeader));	//长连接ping包回报 add by wangtiefeng
194     					}				
195     				}
196     				else
197     				{
198     					m_state = ReadPacket;
199     					ResetRecvBuffer( sizeof( SERVERCOMMANDPACKET ), iNeedRecvLen + sizeof( SERVERCOMMANDPACKET ) );
200     				}    			
201         		}
202         		break;
203         	case ReadPacket:
204         		{
205         			if ( !PacketFinished() ) break;
206         			ParseOnePacket();
207     				m_nLastRecvPinTime = CMyserver::GetInstance()->GetTimeVal().tv_sec;
208     
209         			m_state = ReadPacketHead;
210         			ResetRecvBuffer( 0, sizeof( SERVERCOMMANDPACKET ) );
211         		}
212         		break;
213         	}
214             ii++;
215         }
216     	return 0;
217     }

219     int CCommonTask::SendProcess( void )
220     {
221     	int retValue = 0;
222         int ii=0;
223         while (ii < LONGCONNECT_RECVSNED_NUM)
224         {
225         	if (m_pCurSendMsg != NULL)
226         	{
227         	        retValue = SendData();
228         	}
229         	else
230         	{
231         	        m_pCurSendMsg = GetMsgFromSendList();
232         	        if (m_pCurSendMsg != NULL)
233         	        {
234         	                retValue = SendData();                        
235         	        }                        
236         	        else
237         	        {
238         	        	int tmpEvent = MYPOLLERR|MYPOLLHUP|MYPOLLIN;
239         		        CNetProcessThread *pNetThread = (CNetProcessThread*)m_pThread;
240         		        CEpoll *pEoll = pNetThread->GetEpoll();
241         		        pEoll->ModifyFromEpoll(this, tmpEvent);
242         	            retValue = 1;
243     					break;
244         	        }
245         	}
246             if (retValue == 2 || retValue == -1)
247                  break;		
248             ii++;
249         }
250     
251     	return retValue;
252     }

254     int CCommonTask::PutMsgToSendList( const char *pBuf, int iLen )
255     {
256     	if ( !pBuf )
257     	{
258     		return -1;
259     	}
260     	if ( (int)sizeof(SERVERINNERCOMMANDPACKET) > iLen )
261     	{
262     		return -1;
263     	}
264     	
265     	PSERVERINNERCOMMANDPACKET pServerInnerCommandPacket = ( PSERVERINNERCOMMANDPACKET )pBuf;
266     	if ( pServerInnerCommandPacket->ucDestMachineType != m_ucDestMachineType ||
267     	      pServerInnerCommandPacket->usDestMachineIndex != m_usDestMachineIndex )
268     	{
269     		return -1;
270     	}
271     
272     	WRITE_TRACE( "PutMsgToSendList %d %d", m_ucDestMachineType, m_usDestMachineIndex );
273     	int n = sizeof( pServerInnerCommandPacket->ucDestMachineType ) + sizeof( pServerInnerCommandPacket->usDestMachineIndex );
274     	return SendToList( ( ( char *)pServerInnerCommandPacket ) + n, iLen - n );
275     }

277     int CCommonTask::ProcessPacket( PSERVERCOMMANDPACKET pServerCommandPacket )
278     {
279     	if ( !pServerCommandPacket )
280     	{
281     		return -1;
282     	}
283     	
284     	TMsg tMsg;
285     	tMsg.destEntityType = RECVANDSENDMSGENTITY;
286     	tMsg.destEntityId = ( int )ntohl( pServerCommandPacket->iDestProcessId );
287     	tMsg.srcEntityId = 0;
288     	tMsg.srcEntityType= LONGCONNECTEDENTITY;
289     	tMsg.srcTaskId = GetTaskId();
290     	tMsg.taskId = ( unsigned long long )ntohll( pServerCommandPacket->ullDestTaskId );
291     	tMsg.msgType = CLIENTMSGTYPE;
292     	
293     	int iLength = ( int )ntohl( pServerCommandPacket->iMsgLen);
294     	char *pTmpBuf = new char[iLength];
295     	memcpy( pTmpBuf , pServerCommandPacket->serverCommandPacketContent , iLength );
296     
297     	CMsgBody * pMsgBody = new CClientMsgBody( pTmpBuf , iLength );
298     	
299     	CMsg *pMsg = new CMsg( tMsg, pMsgBody );
300     	
301     	CMyserver::GetInstance()->PutMsgToThread( pMsg );
302     
303     	return 0;
304     }

306     void CCommonTask::ResetRecvBuffer( unsigned int uReceivePos, unsigned int uReceiveSize )
307     {
308     	m_uReceivePos = uReceivePos;
309     	m_uReceiveSize = uReceiveSize;
310     	if ( ( unsigned int )m_vReceiveBuffer.size() < m_uReceiveSize )
311     	{
312     		m_vReceiveBuffer.resize( m_uReceiveSize );
313     	}
314     }

316     int CCommonTask::ParseOnePacket( void )
317     {
318     	PSERVERCOMMANDPACKET pServerCommandPacket = ( PSERVERCOMMANDPACKET )&m_vReceiveBuffer[0];
319     	if ( !pServerCommandPacket )
320     	{
321     		return -1;
322     	}
323     	return ProcessPacket( pServerCommandPacket );
324     }

326     int CCommonTask::AddToEpoll( void )
327     {          
328             int tmpEvent = MYPOLLERR | MYPOLLHUP | MYPOLLIN;                         
329            if (m_sendList.GetNum() != 0 || m_pCurSendMsg != NULL)
330             {
331                     tmpEvent = tmpEvent|MYPOLLOUT;
332             }
333            
334             CNetProcessThread *pNetThread = (CNetProcessThread*)m_pThread;
335             CEpoll *pEoll = pNetThread->GetEpoll();
336             pEoll->AddToEpoll(this, tmpEvent);
337             return 0;
338     }

348     int CCommonTask::SendToList(const char *pBuf, int nLen)
349     {        
350             if (pBuf == NULL || nLen == 0)
351             {       
352                     return -1;
353             }               
354             char *pTmpBuf = new char[nLen];
355             memcpy(pTmpBuf, pBuf, nLen);
356             CSendMsg *pSendMsg = new CSendMsg(nLen, pTmpBuf);
357             
358             m_sendList.push_back(pSendMsg);
359             if (m_sendList.GetNum() == 1) //只有刚刚变化时才修改状态
360             {
361                 int tmpEvent = MYPOLLERR|MYPOLLHUP|MYPOLLIN | MYPOLLOUT;
362                 CNetProcessThread *pNetThread = (CNetProcessThread*)m_pThread;
363                 CEpoll *pEoll = pNetThread->GetEpoll();
364                 pEoll->ModifyFromEpoll(this, tmpEvent);
365             }
366             return 0;
367     }
