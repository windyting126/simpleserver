24     int CMaintainTask::OnTimer(TTimeEvent &timeEvent)
25     {
26     	CTask::OnTimer(timeEvent);
27     	switch(timeEvent.m_nEvent)
28     	{
29     	case STATUPDATETIMEREVENT:
30     		{
31     			WriteRunInfo::WriteLog("Begin print stat info ...");
32     			WriteRunInfo::WriteStat("Now task num is %d", m_nSTaskNum);
33     			CMyserver::GetInstance()->GetCommonStat().Update();                         
34     			AddOneTimer(STATUPDATETIMEREVENT, STATUPDATETIMERLENGTH);
35     			break;
36     		}                
37     	default:
38     		{
39     			WriteRunInfo::WriteLog("maintain task recv invalid timer event %d", timeEvent.m_nEvent);
40     			break;
41     		}
42     	}
43     	return 0;          
44     }
