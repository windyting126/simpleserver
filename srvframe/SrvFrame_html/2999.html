   18 int CThreadGroup::PutMsgToThread(CMsg *pMsg)
   19 {
   20         TMsg *pTMsg = pMsg->GetTMsg();
   21         if (pTMsg->destEntityType != m_iThreadEntityType) //先做简单的判断
   22         {
   23                 //写日志，过滤此消息
   24                 WriteRunInfo::WriteLog("Error entity type(%d) , %d", pTMsg->destEntityType, m_iThreadEntityType);
   25 
   26                 ////直接将此消息删除
   27                 delete pMsg;
   28                 return -1;
   29         }
   30 
   31         if (pTMsg->destEntityId == RANDOMENTITYID) //未指定entityid, 轮循
   32         {                
   33                 m_rotationMutex.Lock();
   34                 m_threadList[m_iRotationEntityId]->PutMsgToInputList(pMsg);               
   35                 m_iRotationEntityId ++;
   36                 m_iRotationEntityId = m_iRotationEntityId%m_iEntityIdSize;
   37                 m_rotationMutex.UnLock();
   38         }
   39         else  //指定entityid
   40         {
   41                 if (pTMsg->destEntityId < 0 || pTMsg->destEntityId >= m_iEntityIdSize) //实体ID范围不正确
   42                 {
   43                         //写日志，过滤此消息
   44                         WriteRunInfo::WriteLog("Error entityid (%d) , %d", pTMsg->destEntityId, m_iEntityIdSize);
   45                         //直接将此消息删除
   46                         delete pMsg;
   47                         return -1;
   48                 }
   49                 //否则就将消息挂入相应的线程中
   50                 m_threadList[pTMsg->destEntityId]->PutMsgToInputList(pMsg);
   51         }
   52         return 0;        
   53 }

   55 int CThreadGroup::Init()
   56 {    
   57         for (int i=0; i < m_iEntityIdSize; i++)
   58         {
   59                 CWorkThread *pThread = CMyserver::GetInstance()->GetClassFactory()->GenWorkThread(m_iThreadEntityType, i, m_sAnnexData, 
   60                         m_arg);
   61                 if (pThread == NULL)
   62                 {
   63                         WriteRunInfo::WriteLog("Malloc memory error");
   64                         return -1;
   65                 }
   66                 int ret = pThread->Start();
   67                 if (ret != 0)
   68                 {
   69                         WriteRunInfo::WriteLog("Start thread fail");
   70                         return -1;
   71                 }
   72                 WriteRunInfo::WriteLog("The threadid is %d, pid is %d", pThread->GetThreadId(), pThread->GetPid());
   73                 m_threadList.push_back(pThread);
   74         } 
   75         //将这个线程组注册到服务器中
   76         CMyserver::GetInstance()->RegisterThreadGroup(this);
   77         return 0;
   78 }

   82 int CThreadGroup::GetRunFlag()
   83 {
   84         int ret = 0;
   85         for (unsigned int i=0; i < m_threadList.size(); i++)
   86         {
   87                 ret += m_threadList[i]->GetRunFlag();
   88         }
   89         return ret;
   90 }

   92 int CThreadGroup::StopAllThread()
   93 {
   94         for (unsigned int i=0; i < m_threadList.size(); i++)
   95         {
   96                 m_threadList[i]->Stop();
   97         }
   98         return 0;
