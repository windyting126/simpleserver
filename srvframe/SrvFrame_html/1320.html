   84 int CEpollReactor::StartService()
   85 {
   86         if (m_bThreadOk)
   87         {
   88                 return 0;       //return 0 for success
   89         }
   90 
   91         m_iEpFd = epoll_create(MAXCLIENT);
   92         if (m_iEpFd <= 0)
   93         {
   94                 return -1;      //return -1 for error(using 0 for error is not a good choice)
   95         }
   96         
   97         if(pthread_create(&m_iThreadId, NULL, CEpollReactor::WorkThreadFunction, this) != 0)
   98         {
   99                 close(m_iEpFd);
  100                 m_iEpFd = -1;
  101                 return -1;      //return -1 for error(using 0 for error is not a good choice)
  102         }
  103         m_bThreadOk = true;
  104 
  105         return 0;               //return 0 for success
  106 }

  128 {
  129         if (strRemoteIp == "")

  133         }
  134         TEpollConnectMsgBody *pMsgBody = new TEpollConnectMsgBody;

  139         pMsgBody->m_strSrvIp = strRemoteIp;

  143         m_msgQueue.Put(pMsg);
  144         WriteRunInfo::WriteLog("LongConnect : Connect [%u,%u] ip=%s,port=%u [%d,%d]", nMachineType, nMachineIndex, strRemoteIp.c_str(), nRemotePort, nEntityType, nEntityId);

  189 void CEpollReactor::WorkProcess()
  190 {
  191         long long nLastSendPing = 0;
  192         while (!m_bIsStop)
  193         {
  194                 CEpollMsg *pMsg = NULL;
  195                 while (m_msgQueue.Get(pMsg))
  196                 {
  197                         if (!pMsg)
  198                         {
  199                                 WriteRunInfo::WriteError("Epoll Reactor : NULL msg");
  200                                 continue;
  201                         }
  202                         Process(pMsg);
  203                         delete pMsg;
  204                 }
  205                 Dispatch();
  206 
  207                 long long nCurTime = CMyserver::GetInstance()->GetTimeVal().tv_sec;
  208                 if (nCurTime - nLastSendPing >= m_nPingInterval)
  209                 {
  210                         nLastSendPing = nCurTime;
  211                         SendPing();
  212                 }
  213         }
  214         return;
  215 }

  217 int CEpollReactor::Process(CEpollMsg *pMsg)
  218 {
  219         int nRet = 0;
  220         switch(pMsg->GetMsgType())
  221         {
  222                 case EPOLLREACTOR_MSG_CONNECT:
  223                         {
  224                                 nRet = ProcessConnect(pMsg);
  225                                 break;
  226                         }
  227                 case EPOLLREACTOR_MSG_RECONNECT:
  228                         {
  229                                 nRet = ProcessReConnect(pMsg);
  230                                 break;
  231                         }
  232                 case EPOLLREACTOR_MSG_DISCONNECT:
  233                         {
  234                                 nRet = ProcessDisConnect(pMsg);
  235                                 break;
  236                         }
  237                 case EPOLLREACTOR_MSG_LOSTCONNECT:
  238                         {
  239                                 nRet = ProcessLostConnect(pMsg);
  240                                 break;
  241                         }
  242                 default:
  243                         {
  244                                 nRet = -1;
  245                                 WriteRunInfo::WriteError("Epoll Rector recv error msg = %d", pMsg->GetMsgType());
  246                                 break;
  247                         }
  248         }
  249         return nRet;
  250 }

  252 int CEpollReactor::ProcessConnect(CEpollMsg *pMsg)
  253 {
  254         TEpollConnectMsgBody *pMsgBody = (TEpollConnectMsgBody*)pMsg->GetMsgBody();
  255         TConDstSrvInfo ConInfo;
  256         ConInfo.m_nServerType = pMsgBody->m_nMachineType;
  257         ConInfo.m_nServerIndex = pMsgBody->m_nMachineIndex;
  258 
  259         
  260         map<TConDstSrvInfo, TConnectItem>::iterator itrMap = m_mapConnect.find(ConInfo);
  261         if (itrMap != m_mapConnect.end())
  262         {
  263                 WriteRunInfo::WriteError("Epoll Reactor Connect : duplicate connect [%u,%u]", ConInfo.m_nServerType, ConInfo.m_nServerIndex);
  264                 return -1;
  265         }
  266         else
  267         {
  268                 map<TConDstSrvInfo, TConnectItem>::iterator itrMapDis = m_mapDisConnect.find(ConInfo);
  269                 if (itrMapDis != m_mapDisConnect.end())
  270                 {
  271                         m_mapDisConnect.erase(itrMapDis);
  272                 }
  273 
  274                 TConnectItem ConnectItem(pMsgBody->m_nMachineType,
  275                                 pMsgBody->m_nMachineIndex,
  276                                 pMsgBody->m_strSrvIp,
  277                                 pMsgBody->m_nSrvPort,
  278                                 pMsgBody->m_nConnectTimeout,
  279                                 pMsgBody->m_nSrcEntityType,
  280                                 pMsgBody->m_nSrcEntityId);
  281                 m_mapConnect.insert(make_pair(ConInfo, ConnectItem));
  282         }
  283         return 0;
  284 }

  286 int CEpollReactor::ProcessReConnect(CEpollMsg *pMsg)
  287 {
  288         TEpollReConnectMsgBody *pMsgBody = (TEpollReConnectMsgBody*)pMsg->GetMsgBody();
  289         TConDstSrvInfo ConInfo;
  290         ConInfo.m_nServerType = pMsgBody->m_nMachineType;
  291         ConInfo.m_nServerIndex = pMsgBody->m_nMachineIndex;
  292         map<TConDstSrvInfo, TConnectItem>::iterator itrMap = m_mapConnect.find(ConInfo);
  293         if (itrMap == m_mapConnect.end())
  294         {
  295                 WriteRunInfo::WriteError("Epoll Reactor ReConnect : not find [%u,%u]", ConInfo.m_nServerType, ConInfo.m_nServerIndex);
  296                 return -1;
  297         }
  298         itrMap->second.m_sRemoteIp = pMsgBody->m_strSrvIp;
  299         itrMap->second.m_usRemotePort = pMsgBody->m_nSrvPort;
  300         itrMap->second.m_ullExpireTime = pMsgBody->m_nConnectTimeout;
  301 
  302         if (itrMap->second.m_nState == TConnectItem::EIS_ESTABLISH)
  303         {
  304                 SendDestoryConnectMsg(itrMap->second.m_ucMachineType, itrMap->second.m_usMachineIndex, itrMap->second.m_iDestEntityType, itrMap->second.m_iDestEntityId);
  305         }
  306         else
  307         {
  308                 if (itrMap->second.m_nSocketId >= 0)
  309                 {
  310                         struct epoll_event ev = {0, {0}};
  311                         ev.data.ptr = &(itrMap->second);
  312                         epoll_ctl(m_iEpFd, EPOLL_CTL_DEL, itrMap->second.m_nSocketId, &ev);
  313                         close(itrMap->second.m_nSocketId);
  314                         itrMap->second.m_nSocketId = -1;
  315                 }
  316                 itrMap->second.m_nTryConnTime = GetTime(0);
  317         }
  318         return 0;
  319 }

  321 int CEpollReactor::ProcessDisConnect(CEpollMsg *pMsg) 
  322 {
  323         TEpollLostConnectMsgBody *pMsgBody = (TEpollLostConnectMsgBody*)pMsg->GetMsgBody();
  324         TConDstSrvInfo ConInfo;
  325         ConInfo.m_nServerType = pMsgBody->m_nMachineType;
  326         ConInfo.m_nServerIndex = pMsgBody->m_nMachineIndex;
  327 
  328         map<TConDstSrvInfo, TConnectItem>::iterator itrMapDis = m_mapDisConnect.find(ConInfo);
  329         if (itrMapDis != m_mapDisConnect.end())
  330         {
  331                 WriteRunInfo::WriteError("Epoll Reactor DisConnect : duplicate disconnect [%u,%u]", ConInfo.m_nServerType, ConInfo.m_nServerIndex);
  332                 return -1;
  333         }
  334 
  335         map<TConDstSrvInfo, TConnectItem>::iterator itrMap = m_mapConnect.find(ConInfo);
  336         if (itrMap == m_mapConnect.end())
  337         {
  338                 WriteRunInfo::WriteError("Epoll Reactor DisConnect : not find connect [%u,%u]", ConInfo.m_nServerType, ConInfo.m_nServerIndex);
  339                 return -1;
  340         }
  341         if (itrMap->second.m_nState == TConnectItem::EIS_ESTABLISH)
  342         {
  343                 SendDestoryConnectMsg(itrMap->second.m_ucMachineType, itrMap->second.m_usMachineIndex, itrMap->second.m_iDestEntityType, itrMap->second.m_iDestEntityId);
  344                 m_mapDisConnect.insert(make_pair(ConInfo, itrMap->second));
  345                 m_mapConnect.erase(itrMap);
  346         }
  347         else
  348         {
  349                 if (itrMap->second.m_nSocketId >= 0)
  350                 {
  351                         struct epoll_event ev = {0, {0}};
  352                         ev.data.ptr = &(itrMap->second);
  353                         epoll_ctl(m_iEpFd, EPOLL_CTL_DEL, itrMap->second.m_nSocketId, &ev);
  354                         close(itrMap->second.m_nSocketId);
  355                 }
  356                 m_mapConnect.erase(itrMap);
  357         }
  358         return 0;
  359 }

  361 int CEpollReactor::ProcessLostConnect(CEpollMsg *pMsg) 
  362 {
  363         TEpollLostConnectMsgBody *pMsgBody = (TEpollLostConnectMsgBody*)pMsg->GetMsgBody();
  364         TConDstSrvInfo ConInfo;
  365         ConInfo.m_nServerType = pMsgBody->m_nMachineType;
  366         ConInfo.m_nServerIndex = pMsgBody->m_nMachineIndex;
  367         
  368         map<TConDstSrvInfo, TConnectItem>::iterator itrMap = m_mapConnect.find(ConInfo);
  369         if (itrMap != m_mapConnect.end())
  370         {
  371                 itrMap->second.m_nState = TConnectItem::EIS_INIT;
  372                 itrMap->second.m_nTryConnTime = GetTime(0);
  373                 itrMap->second.m_nSocketId = -1;
  374         }
  375         else
  376         {
  377                 map<TConDstSrvInfo, TConnectItem>::iterator itrMapDis = m_mapDisConnect.find(ConInfo);
  378                 if (itrMapDis != m_mapDisConnect.end())
  379                 {
  380                         m_mapDisConnect.erase(itrMapDis);               
  381                         WriteRunInfo::WriteLog("Epoll Reactor DisConnect : success [%u,%u]", ConInfo.m_nServerType, ConInfo.m_nServerIndex);
  382                 }
  383                 else
  384                 {
  385                         WriteRunInfo::WriteError("Epoll Reactor Lost Connect : not find connect [%u,%u]", ConInfo.m_nServerType, ConInfo.m_nServerIndex);
  386                         return -1;
  387                 }
  388         }
  389         return 0;
  390 }

  400 int CEpollReactor::SendPing()
  401 {
  402         map<TConDstSrvInfo, TConnectItem>::iterator itrMap = m_mapConnect.begin();
  403         for (; itrMap != m_mapConnect.end(); itrMap ++)
  404         {
  405                 if (itrMap->second.m_nState == TConnectItem::EIS_ESTABLISH)
  406                 {
  407                         SendPing(itrMap->second.m_ucMachineType, itrMap->second.m_usMachineIndex);
  408                 }
  409         }
  410         return 0;
  411 }

  413 int CEpollReactor::CheckTimeout()
  414 {
  415         map<TConDstSrvInfo, TConnectItem>::iterator itrMap = m_mapConnect.begin();
  416         for (; itrMap != m_mapConnect.end(); itrMap ++)
  417         {
  418                 if ((itrMap->second.m_nState == TConnectItem::EIS_ESTABLISH) || (itrMap->second.m_nSocketId < 0))
  419                 {
  420                         continue;
  421                 }
  422                 else if (GetTime(0) - itrMap->second.m_nTryConnTime > itrMap->second.m_ullExpireTime)
  423                 {
  424                         struct epoll_event ev = {0, {0}};
  425                         ev.data.ptr = &(itrMap->second);
  426                         epoll_ctl(m_iEpFd, EPOLL_CTL_DEL, itrMap->second.m_nSocketId, &ev);
  427                         close(itrMap->second.m_nSocketId);
  428                         itrMap->second.m_nSocketId = -1;
  429                         itrMap->second.m_nTryConnTime = GetTime(itrMap->second.m_ullExpireTime);
  430 
  431                         WriteRunInfo::WriteLog("LongConnect : Timeout [%u,%u] ip=%s,port=%u [%d,%d]", itrMap->second.m_ucMachineType, itrMap->second.m_usMachineIndex, itrMap->second.m_sRemoteIp.c_str(), itrMap->second.m_usRemotePort, itrMap->second.m_iDestEntityType, itrMap->second.m_iDestEntityId);
  432                 }
  433         }
  434         return 0;
  435 }

  437 int CEpollReactor::DispatchConnect()
  438 {
  439         map<TConDstSrvInfo, TConnectItem>::iterator itrMap = m_mapConnect.begin();
  440         for (; itrMap != m_mapConnect.end(); itrMap ++)
  441         {
  442                 if (itrMap->second.m_nState != TConnectItem::EIS_INIT || itrMap->second.m_nSocketId >= 0)
  443                 {
  444                         continue;
  445                 }
  446                 if (itrMap->second.m_nTryConnTime > GetTime(0))
  447                 {
  448                         continue;
  449                 }
  450                 itrMap->second.m_nSocketId = socket(AF_INET, SOCK_STREAM, 0);
  451                 if (itrMap->second.m_nSocketId < 0)
  452                 {
  453                         itrMap->second.m_nTryConnTime = GetTime(itrMap->second.m_ullExpireTime);
  454                 }
  455                 unsigned long nIp;
  456                 if (SetNoBlock(itrMap->second.m_nSocketId) != 0 || inet_pton(AF_INET, itrMap->second.m_sRemoteIp.c_str(), &nIp) <= 0) 
  457                 {
  458                         close(itrMap->second.m_nSocketId);
  459                         itrMap->second.m_nSocketId = -1;
  460                         itrMap->second.m_nTryConnTime = GetTime(itrMap->second.m_ullExpireTime);
  461                         continue;
  462                 }
  463                 itrMap->second.m_nTryConnTime = GetTime(0);
  464 
  465                 sockaddr_in servaddr;
  466                 memset(&servaddr , 0, sizeof(servaddr));
  467                 servaddr.sin_family = AF_INET;
  468                 servaddr.sin_addr.s_addr = nIp;
  469                 servaddr.sin_port = htons(itrMap->second.m_usRemotePort);
  470 
  471                 connect(itrMap->second.m_nSocketId, (struct sockaddr*)&servaddr, sizeof(servaddr));
  472 
  473                 epoll_event ev = {0, {0}};
  474                 ev.data.ptr = &(itrMap->second);
  475                 ev.events = EPOLLOUT;
  476                 epoll_ctl(m_iEpFd, EPOLL_CTL_ADD, itrMap->second.m_nSocketId, &ev);
  477 
  478                 WriteRunInfo::WriteLog("LongConnect : Start Connect [%u,%u] ip=%s,port=%u [%d,%d]", itrMap->second.m_ucMachineType, itrMap->second.m_usMachineIndex, itrMap->second.m_sRemoteIp.c_str(), itrMap->second.m_usRemotePort, itrMap->second.m_iDestEntityType, itrMap->second.m_iDestEntityId);
  479         }
  480         return 0;
  481 }

  483 int CEpollReactor::EpollWait()
  484 {
  485         epoll_event events[MAXCLIENT];
  486         int nFdNum = epoll_wait(m_iEpFd, events, MAXCLIENT, SCAN_TIME);
  487         if (nFdNum < 0)
  488         {
  489                 if(EINTR == errno) return 0;
  490                 WriteRunInfo::WriteError("epoll_wait return -1 error %d", errno);
  491                 return -1;
  492         }
  493         for (int nA = 0; nA < nFdNum; nA ++)
  494         {
  495                 int nSocketId = ((TConnectItem *)(events[nA].data.ptr))->m_nSocketId;
  496                 TConnectItem *pConnectItem = (TConnectItem *)events[nA].data.ptr;
  497                 
  498                 //delete from epoll
  499                 struct epoll_event ev = {0, {0}};
  500                 ev.data.ptr = pConnectItem;
  501                 epoll_ctl(m_iEpFd, EPOLL_CTL_DEL, nSocketId, &ev);
  502 
  503                 if (events[nA].events & EPOLLOUT)
  504                 {
  505                         int nError;
  506                         socklen_t nSockLen = sizeof(nError);
  507                         if ((getsockopt(pConnectItem->m_nSocketId, SOL_SOCKET, SO_ERROR, &nError, &nSockLen) == 0) && (nError == 0))
  508                         {
  509                                 pConnectItem->m_nState = TConnectItem::EIS_ESTABLISH;
  510                                 SendConnectMsg(pConnectItem->m_nSocketId,
  511                                                 pConnectItem->m_sRemoteIp.c_str(), pConnectItem->m_usRemotePort,
  512                                                 pConnectItem->m_ucMachineType, pConnectItem->m_usMachineIndex,
  513                                                 pConnectItem->m_iDestEntityType, pConnectItem->m_iDestEntityId);
  514                                 WriteRunInfo::WriteLog("LongConnect : ConnectSuc [%u,%u] ip=%s,port=%u [%d,%d]", pConnectItem->m_ucMachineType, pConnectItem->m_usMachineIndex, pConnectItem->m_sRemoteIp.c_str(), pConnectItem->m_usRemotePort, pConnectItem->m_iDestEntityType, pConnectItem->m_iDestEntityId);
  515                         }
  516                         else
  517                         {
  518                                 close(pConnectItem->m_nSocketId);
  519                                 pConnectItem->m_nSocketId = -1;
  520                                 pConnectItem->m_nTryConnTime = GetTime(pConnectItem->m_ullExpireTime);
  521                                 WriteRunInfo::WriteLog("LongConnect : ConnectFail [%u,%u] ip=%s,port=%u [%d,%d]", pConnectItem->m_ucMachineType, pConnectItem->m_usMachineIndex, pConnectItem->m_sRemoteIp.c_str(), pConnectItem->m_usRemotePort, pConnectItem->m_iDestEntityType, pConnectItem->m_iDestEntityId);
  522                         }
  523                 }
  524                 else
  525                 {
  526                         close(pConnectItem->m_nSocketId);
  527                         pConnectItem->m_nSocketId = -1;
  528                         pConnectItem->m_nTryConnTime = GetTime(pConnectItem->m_ullExpireTime);
  529                         WriteRunInfo::WriteLog("LongConnect : ConnectFail [%u,%u] ip=%s,port=%u [%d,%d]", pConnectItem->m_ucMachineType, pConnectItem->m_usMachineIndex, pConnectItem->m_sRemoteIp.c_str(), pConnectItem->m_usRemotePort, pConnectItem->m_iDestEntityType, pConnectItem->m_iDestEntityId);
  530                 }
  531         }

  589         serverInnerCommandPacket.serverCommandPacket.iMsgLen = 0;

  592
