38     int CLongConnSendAndRecvThread::Process(CMsg* pMsg)
39     {
40     	TMsg* pTMsg = pMsg->GetTMsg();
41     
42     	if(pTMsg->msgType == MSGTYPE_LONGCONN_COMMON)
43     	{
44     		CLongConnBufMsgBody* pMsgBody = (CLongConnBufMsgBody*)pMsg->GetMsgBody();
45     		if (!pMsgBody)
46     		{
47     			return 0;
48     		}	
49     		CSocketTask* pSocketTask = FindSocketTask(pTMsg->taskId);
50     		if (pSocketTask)
51     		{
52     			pSocketTask->PutMsgToSendList(pMsgBody->GetBuf(), pMsgBody->GetBufLen());
53     			pMsgBody->DetachBuf();
54     		}
55     	}
56     	else if (pTMsg->msgType == MSGTYPE_LONGCONN_CONNECT)
57     	{
58     		int nRet;
59     		CLongConnectMsgBody *pMsgBody = (CLongConnectMsgBody*)pMsg->GetMsgBody();
60     		if (!pMsgBody)
61     		{
62     			return 0;
63     		}						
64     		TLongConnectOption& LongConnectOption = pMsgBody->GetConnectOption();
65     		char szConnIp[INET_ADDRSTRLEN];
66     		int nIp = htonl(LongConnectOption.m_nSrvIp);
67     		inet_ntop(AF_INET, &nIp, szConnIp, sizeof(szConnIp));
68     		WriteRunInfo::WriteLog("Connect [type=%d,index=%d],[ip=%s,port=%d]", LongConnectOption.m_nLongConnType, LongConnectOption.m_nLongConnIndex, szConnIp, LongConnectOption.m_nSrvPort);
69     		CLongConnSendAndRecvTask* pSocketTask = (CLongConnSendAndRecvTask*)CMyserver::GetInstance()->GetClassFactory()->GenSocketTask(this, NULL, LONGCONNSENDANDRECVTASK);
70     		pSocketTask->Init();
71     		nRet = pSocketTask->ConnectServer(LongConnectOption);
72     		if (nRet != 0)
73     		{
74     			pSocketTask->SendConnectState(TLongConnectState::CCR_CREATESOCKET);
75     			delete pSocketTask;
76     		}
77     	}
78     	else if (pTMsg->msgType == MSGTYPE_LONGCONN_DISCONNECT)
79     	{
80     		CSocketTask* pSocketTask = FindSocketTask(pTMsg->taskId);
81     		if (pSocketTask)
82     		{
83     			delete pSocketTask;
84     		}
85     	}
86     	return 0;	
87     }

92     	return 0;
93     }

95     int CLongConnSendAndRecvThread::CheckConnectValidity()
96     {
97     	time_t nCurTime = CMyserver::GetInstance()->GetTimeVal().tv_sec;
98     	if (nCurTime > m_nLastCheckConnectTime)
99     	{
100     		int nRet = 0;
101     		map<ULONG64, CSocketTask*>::iterator itrMap = m_pMapSocketList.begin();
102     		while (itrMap != m_pMapSocketList.end())
103     		{
104     			map<ULONG64, CSocketTask*>::iterator itrCur = itrMap;
105     			itrMap ++;
106     			nRet += (itrCur->second)->CheckTimeOut();
107     		}
108     		m_nLastCheckConnectTime = nCurTime;
109     	}
110     	return 0;
111     
112     }
