   40 int CHttpRecvAndSendTaskProcess::ProcessRecvPacket(const char *pPacket, unsigned int uiBufLen)
   41 {
   42         int nLen = uiBufLen;
   43         if(uiBufLen <= 0)
   44                 return -1;
   45         char *pBuf = new char[nLen];
   46         memcpy(pBuf, pPacket, nLen);
   47         CBufMsgBody *pBody = new CBufMsgBody(pBuf, nLen);
   48         m_pNetProcessThread->SendMsg(PROCESSENTITY, -1, -1, CLIENTMSGTYPE, pBody, m_pRecvAndSendTask->GetTaskId());
   49         return 0;
   50 }
