54     char* CLongConnMgr::GenResPacket(char* pBuf, int iBufLen, TFrameLongConnectHead* pHead, int& iOutBufLen, int nLongConnMsgId /* = LONGCONN_CMDID_S2C */)
55     {
56     	if (!pBuf || iBufLen <= 0 || !pHead)
57     	{
58     		return NULL;
59     	}
60     	int iTmpBufLen = TFrameLongConnectHead::FLCH_HEADLEN + iBufLen;
61     	char* pTmpBuf = new char[iTmpBufLen];
62     	CBufMgr BufMgr(pTmpBuf, iTmpBufLen);
63     	TFrameLongConnectHead CmdHead;
64     	CmdHead.m_nVersion = pHead->m_nVersion;
65     	CmdHead.m_nSequence = pHead->m_nSequence;
66     	CmdHead.m_nCmdId = nLongConnMsgId;
67     	CmdHead.m_nSrcEntityType = pHead->m_nDstEntityType;
68     	CmdHead.m_nSrcEntityId = pHead->m_nDstEntityId;
69     	CmdHead.m_nSrcTaskId = pHead->m_nDstTaskId;
70     	CmdHead.m_nDstEntityType = pHead->m_nSrcEntityType;
71     	CmdHead.m_nDstEntityId = pHead->m_nSrcEntityId;
72     	CmdHead.m_nDstTaskId = pHead->m_nSrcTaskId;
73     	CmdHead.m_nBodyLen = iBufLen;
74     	CmdHead.Encode(BufMgr);
75     	BufMgr.PutCharBuf(pBuf, iBufLen);
76     	BufMgr.DistachBuf();
77     	iOutBufLen = iTmpBufLen;
78     	return pTmpBuf;
79     }

105     int CLongConnMgr::GetConnectState(int iType, int iIndex)
106     {
107     	if(iType < 0 ||  iType >= MAX_LONGCONTYPE_SIZE || iIndex < 0 || iIndex >= MAX_LONGCONINDEX_SIZE)
108     	{
109     		WriteRunInfo::WriteLog("CLongConnMgr::GetConnectState : Long Conn Type[%d] or Index[%d] out of range", iType, iIndex);
110     		return -1;
111     	}
112     	int nState = TLongConnectState::CCR_INIT;
113     	m_arrLongConnLock[iType][iIndex].GetReadLock();
114     	nState = m_arrLongConnState[iType][iIndex].m_nResult;
115     	m_arrLongConnLock[iType][iIndex].ReleaseLock();
116     	return nState;
117     }

119     int CLongConnMgr::SetEntitySize(int iEntityType, int iEntitySize)
120     {
121     	if (iEntityType < 0 || iEntitySize <= 0)
122     	{
123     		return -1;
124     	}
125     	CThreadLock ThreadLock(&m_lockEntityLongConnInfo);
126     	map<int, TEntityLongConnInfo>::iterator itrMap = m_mapEntityLongConnInfo.find(iEntityType);
127     	if (itrMap != m_mapEntityLongConnInfo.end())
128     	{
129     		return -1;
130     	}
131     	TEntityLongConnInfo EntityLongConnInfo;
132     	EntityLongConnInfo.m_nEntityType = iEntityType;
133     	EntityLongConnInfo.m_nEntitySize = iEntitySize;
134     	EntityLongConnInfo.m_vecLongConnNum.reserve(iEntitySize);
135     	for (int nA = 0; nA < iEntitySize; nA ++)
136     	{
137     		EntityLongConnInfo.m_vecLongConnNum.push_back(0);	
138     	}
139     	m_mapEntityLongConnInfo.insert(make_pair(iEntityType, EntityLongConnInfo));
140     	return 0;
141     }

143     int CLongConnMgr::GetEntityId(int iEntityType)
144     {
145     	if (iEntityType < 0)
146     	{
147     		return RANDOMENTITYID;
148     	}
149     	CThreadLock ThreadLock(&m_lockEntityLongConnInfo);
150     	map<int, TEntityLongConnInfo>::iterator itrMap = m_mapEntityLongConnInfo.find(iEntityType);
151     	if (itrMap == m_mapEntityLongConnInfo.end())
152     	{
153     		return RANDOMENTITYID;
154     	}
155     	int iEntitiId = 0;
156     	int iMinConnectNum = itrMap->second.m_vecLongConnNum[0];
157     	for (int nA = 1; nA < itrMap->second.m_nEntitySize; nA ++)
158     	{
159     		if (itrMap->second.m_vecLongConnNum[nA] < iMinConnectNum)
160     		{
161     			iMinConnectNum = itrMap->second.m_vecLongConnNum[nA];
162     			iEntitiId = nA;
163     		}
164     	}
165     	itrMap->second.m_vecLongConnNum[iEntitiId] ++;
166     	return iEntitiId;
167     }

169     int CLongConnMgr::DelEntityId(int iEntityType, int iEntityId)
170     {
171     	if (iEntityType < 0 || iEntityId < 0)
172     	{
173     		return -1;
174     	}
175     	CThreadLock ThreadLock(&m_lockEntityLongConnInfo);
176     	map<int, TEntityLongConnInfo>::iterator itrMap = m_mapEntityLongConnInfo.find(iEntityType);
177     	if (itrMap == m_mapEntityLongConnInfo.end())
178     	{
179     		return -1;
180     	}
181     	if (iEntityId >= itrMap->second.m_nEntitySize)
182     	{
183     		return -1;
184     	}
185     	itrMap->second.m_vecLongConnNum[iEntityId] --;
186     	return 0;
187     }

189     int CLongConnMgr::CreateConnect(const TLongConnectOption& ConnectOption)
190     {
191     	int iType = ConnectOption.m_nLongConnType;
192     	int iIndex = ConnectOption.m_nLongConnIndex;
193     	int iEntityType = ConnectOption.m_nDstEntityType;
194     	if (GetConnectState(iType, iIndex) != TLongConnectState::CCR_INIT)
195     	{
196     		WriteRunInfo::WriteLog("CLongConnMgr::CreateConnect : duplicate create connect[%d:%d]", iType, iIndex);
197     		return -1;
198     	}
199     
200     	m_arrLongConnLock[iType][iIndex].GetWriteLock();
201     
202     	m_arrLongConnOption[iType][iIndex] = ConnectOption;
203     	m_arrLongConnState[iType][iIndex].m_nEntityType = iEntityType;
204     	m_arrLongConnState[iType][iIndex].m_nEntityId = GetEntityId(iEntityType);
205     	
206     	SendConnect(iType, iIndex);
207     
208     	m_arrLongConnLock[iType][iIndex].ReleaseLock();
209     
210     	WriteRunInfo::WriteLog("CLongConnMgr::CreateConnect : Connect [%d][%d]", iType, iIndex);
211     	return 0;
212     }

214     int CLongConnMgr::DestoryConnect(int iType, int iIndex)
215     {
216     	if (GetConnectState(iType, iIndex) < 0)	
217     	{
218     		WriteRunInfo::WriteLog("CLongConnMgr::DestoryConnect : GetConnectState[%d][%d] fail", iType, iIndex);
219     		return -1;
220     	}
221     
222     	m_arrLongConnLock[iType][iIndex].GetWriteLock();
223     
224     	m_pThread->SendMsg(m_arrLongConnState[iType][iIndex].m_nEntityType, m_arrLongConnState[iType][iIndex].m_nEntityId, m_arrLongConnState[iType][iIndex].m_nTaskId, MSGTYPE_LONGCONN_DISCONNECT, NULL, 0);	
225     	DelEntityId(m_arrLongConnState[iType][iIndex].m_nEntityType, m_arrLongConnState[iType][iIndex].m_nEntityId);
226     	memset(&m_arrLongConnOption[iType][iIndex], 0, sizeof(TLongConnectOption));
227     	memset(&m_arrLongConnState[iType][iIndex], 0, sizeof(TLongConnectState));
228     
229     	m_arrLongConnLock[iType][iIndex].ReleaseLock();
230     
231     	WriteRunInfo::WriteLog("CLongConnMgr::DestoryConnect : DisConnect[%d][%d]", iType, iIndex);
232     	return 0;
233     }

235     int CLongConnMgr::ReConnect(const TLongConnectOption& ConnectOption)
236     {
237     	int iType = ConnectOption.m_nLongConnType;
238     	int iIndex = ConnectOption.m_nLongConnIndex;
239     	DestoryConnect(iType, iIndex);

244     int CLongConnMgr::SendLongConnectMsg(CWorkThread* pThread, int iType, int iIndex, CLongConnBufMsgBody* pMsgBody, ULONG64 ulSrcTaskId)
245     {
246     	if (iType < 0 || iType >= MAX_LONGCONTYPE_SIZE || iIndex < 0 || iIndex >= MAX_LONGCONINDEX_SIZE)
247     	{
248     		WriteRunInfo::WriteLog("CLongConnMgr::SendLongConnectMsg : wrong type or index [%d:%d]", iType, iIndex);
249     		return -1;
250     	}
251     
252     	int nRet = 0;
253     	m_arrLongConnLock[iType][iIndex].GetReadLock();
254     	if (m_arrLongConnState[iType][iIndex].m_nResult ==  TLongConnectState::CCR_CONNECTSUC)
255     	{
256     		pThread->SendMsg(m_arrLongConnState[iType][iIndex].m_nEntityType, m_arrLongConnState[iType][iIndex].m_nEntityId, m_arrLongConnState[iType][iIndex].m_nTaskId, MSGTYPE_LONGCONN_COMMON, pMsgBody, ulSrcTaskId);	
257     		nRet = 1;
258     	}
259     	m_arrLongConnLock[iType][iIndex].ReleaseLock();
260     	return nRet;
261     }

263     int CLongConnMgr::SendLongConnectMsg(CWorkThread* pThread, int iType, int iIndex, ULONG64 ulSrcTaskId, char* pBuf, int iBufLen, int nLongConnMsgId /*  = LONGCONN_CMDID_C2S */, int nVersion /*  = 0 */, int nSequenceNum /* = 0 */)
264     {
265     	if (iType < 0 || iType >= MAX_LONGCONTYPE_SIZE || iIndex < 0 || iIndex >= MAX_LONGCONINDEX_SIZE || !pBuf || iBufLen <= 0)
266     	{
267     		WriteRunInfo::WriteLog("CLongConnMgr::SendLongConnectMsg : wrong type or index [%d:%d]", iType, iIndex);
268     		return -1;
269     	}
270     	int nRet = 0;
271     	m_arrLongConnLock[iType][iIndex].GetReadLock();
272     	if (m_arrLongConnState[iType][iIndex].m_nResult ==  TLongConnectState::CCR_CONNECTSUC)
273     	{
274     		int iTmpBufLen = TFrameLongConnectHead::FLCH_HEADLEN + iBufLen;
275     		char* pTmpBuf = new char[iTmpBufLen];
276     		CBufMgr BufMgr(pTmpBuf, iTmpBufLen);
277     		TFrameLongConnectHead CmdHead;
278     		memset(&CmdHead, 0, sizeof(CmdHead));
279     		CmdHead.m_nVersion = nVersion;
280     		CmdHead.m_nSequence = nSequenceNum;
281     		CmdHead.m_nCmdId = nLongConnMsgId;
282     		CmdHead.m_nSrcEntityType = pThread->GetEntityType();
283     		CmdHead.m_nSrcEntityId = pThread->GetEntityId();
284     		CmdHead.m_nSrcTaskId = ulSrcTaskId;
285     		CmdHead.m_nDstEntityType = m_arrLongConnState[iType][iIndex].m_nEntityType;
286     		CmdHead.m_nDstEntityId = m_arrLongConnState[iType][iIndex].m_nEntityId;
287     		CmdHead.m_nDstTaskId = m_arrLongConnState[iType][iIndex].m_nTaskId;
288     		CmdHead.m_nBodyLen = iBufLen;
289     		CmdHead.Encode(BufMgr);
290     		BufMgr.PutCharBuf(pBuf, iBufLen);
291     
292     		CLongConnBufMsgBody* pMsgBody = new CLongConnBufMsgBody(BufMgr.GetBuf(), BufMgr.GetBufLen());	
293     		BufMgr.DistachBuf();
294     
295     		pThread->SendMsg(m_arrLongConnState[iType][iIndex].m_nEntityType, m_arrLongConnState[iType][iIndex].m_nEntityId, m_arrLongConnState[iType][iIndex].m_nTaskId, MSGTYPE_LONGCONN_COMMON, pMsgBody, ulSrcTaskId);	
296     		nRet = 1;
297     	}
298     	m_arrLongConnLock[iType][iIndex].ReleaseLock();
299     	return nRet;
300     }

302     int CLongConnMgr::OnTime()
303     {
304     	int nRet = 0;
305     	for(int i = 0; i < MAX_LONGCONTYPE_SIZE; i++)
306     	{
307     		for(int j = 0; j < MAX_LONGCONINDEX_SIZE; j++)
308     		{
309     			bool bIfUsed = false;
310     			m_arrLongConnLock[i][j].GetReadLock(); 
311     			if (m_arrLongConnOption[i][j].m_nSrvIp && m_arrLongConnOption[i][j].m_nSrvPort)	
312     			{
313     				bIfUsed = true;	
314     			}
315     			m_arrLongConnLock[i][j].ReleaseLock();
316     			if (bIfUsed)
317     			{
318     				OnTimeProcess(i, j);
319     			}
320     		}
321     	}
322     	return nRet;
323     }

325     int CLongConnMgr::OnTimeProcess(int iType, int iIndex)
326     {
327     	m_arrLongConnLock[iType][iIndex].GetReadLock(); 
328     	if (m_arrLongConnState[iType][iIndex].m_nResult == TLongConnectState::CCR_CONNECTSUC)	
329     	{
330     		SendPing(iType, iIndex);
331     	}
332     	else if (m_arrLongConnState[iType][iIndex].m_nResult != TLongConnectState::CCR_WAIT)
333     	{
334     		SendConnect(iType, iIndex);
335     	}
336     	m_arrLongConnLock[iType][iIndex].ReleaseLock();
337     	return 0;
338     }

356     int CLongConnMgr::SendConnect(int iType, int iIndex)
357     {
358     	m_arrLongConnOption[iType][iIndex].m_nSequence = ++ m_nSequence;
359     	CLongConnectMsgBody* pMsgBody = new CLongConnectMsgBody(m_arrLongConnOption[iType][iIndex]);
360     	m_pThread->SendMsg(m_arrLongConnState[iType][iIndex].m_nEntityType, m_arrLongConnState[iType][iIndex].m_nEntityId, 0, MSGTYPE_LONGCONN_CONNECT, pMsgBody, 0);
361     	return 0;
362     }

364     int CLongConnMgr::OnConnectState(const TLongConnectOption& ConnectOption, const TLongConnectState& ConnectState)
365     {
366     	m_arrLongConnLock[ConnectOption.m_nLongConnType][ConnectOption.m_nLongConnIndex].GetWriteLock(); 
367     	if (ConnectOption.m_nSequence == m_arrLongConnOption[ConnectOption.m_nLongConnType][ConnectOption.m_nLongConnIndex].m_nSequence)
368     	{
369     		m_arrLongConnState[ConnectOption.m_nLongConnType][ConnectOption.m_nLongConnIndex] = ConnectState;
370     	}
371     	m_arrLongConnLock[ConnectOption.m_nLongConnType][ConnectOption.m_nLongConnIndex].ReleaseLock();
372     	return 0;
373     }
