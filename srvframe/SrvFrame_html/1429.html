19     CTask *CWorkThread::FindTask(ULONG64 iTaskId)
20     {
21             CTask *pTmpTask = NULL;
22             map<ULONG64, CTask*>::iterator itr = m_taskList.find(iTaskId);
23             if (itr != m_taskList.end())
24             {
25                     pTmpTask = itr->second;
26             }
27             return pTmpTask;        
28     }

42     int CWorkThread::TimerMsgProcess(CMsg *pMsg)
43     {
44             TMsg *pTMsg = pMsg->GetTMsg();
45             CTask *pTask = FindTask(pTMsg->taskId);
46             if (pTask != NULL)
47             {
48                     CBufMsgBody *pBufMsg = (CBufMsgBody*)pMsg->GetMsgBody();
49                     TTimeEvent *pTimerEv = (TTimeEvent*)pBufMsg->GetBuf();
50                     pTask->OnTimer(*pTimerEv);
51             }
52             else
53             {
54                     WriteRunInfo::WriteLog("The task(%llu) is not exist", pTMsg->taskId);
55             }
56             return 0;
57     }
