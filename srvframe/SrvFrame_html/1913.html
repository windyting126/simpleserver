29     int CUdpSendThread::Init()
30     {
31     	if (m_UdpSocket.GetSocket() == INVALID_SOCKET)
32     	{
33     		WriteRunInfo::WriteLog("UdpSendThread::Init : CreateUdpSocket fail");
34     		return -1;
35     	}
36     	return 0;
37     }

39     int CUdpSendThread::Run()
40     {
41     	while (m_nRunFlag)
42     	{
43     		CMsg *pMsg = m_InputList.get();
44     		if (!pMsg) continue;
45     		Process(pMsg);
46     		delete pMsg;
47     	}
48     	return 0;
49     }
50     
51     int CUdpSendThread::PutMsgToInputList(CMsg* pMsg)
52     {
53     	m_InputList.put(pMsg);
54     	return 0;
55     }
56     
57     int CUdpSendThread::GetMsgQueueLen()
58     {
59     	return m_InputList.size();
60     }
61     
62     int CUdpSendThread::Process(CMsg* pMsg)
63     {
64     	CUdpCommonBufMsgBody* pMsgBody = (CUdpCommonBufMsgBody*)pMsg->GetMsgBody();
65     	if (pMsgBody && pMsgBody->GetBuf() && pMsgBody->Size() > 0)
66     	{
67     		m_UdpSocket.UdpSend(pMsgBody->GetBuf(), pMsgBody->Size(), pMsgBody->GetClientIp(), pMsgBody->GetCientPort());
68     	}
69     	return 0;
70     }
