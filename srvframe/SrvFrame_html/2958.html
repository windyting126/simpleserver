46     CWorkThread * CClassFactory::GenWorkThread(int nEntityType, int nEntityId, string sAnnexData, void *arg)
47     {
48     	CWorkThread *pWorkThread = NULL;
49     	switch(nEntityType)
50     	{
51     	case TIMERENTITY:
52     		{
53     			pWorkThread = new CTimerThread(nEntityType, nEntityId);
54     			break;
55     		}
56     	case LISTENENTITY:
57     		{
58     			int nPort;
59     			memcpy(&nPort, sAnnexData.c_str(), 4);
60     			string sHost = sAnnexData.substr(4); 
61     			TListenBlock *pListenBlock = NULL;
62     			if (arg == NULL)
63     				pListenBlock = new TListenBlock;
64     			else
65     				pListenBlock = (TListenBlock*)arg;
66     			pWorkThread = new CListenThread(nEntityType, nEntityId, sHost, nPort, pListenBlock);
67     			break;
68     		}
69     	case RECVANDSENDMSGENTITY:
70     		{
71     			pWorkThread = new CRecvAndSendMsgThread(nEntityType, nEntityId);
72     			break;
73     		}
74     	case MAINTAINENTITY:
75     		{
76     			pWorkThread = new CMaintainThread(nEntityType, nEntityId);
77     			break;
78     		}
79     	
80     
81     	default:
82     		{                        
83     			break;
84     		}
85     	}          
86     	return pWorkThread;
87     }
