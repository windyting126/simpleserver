   29 int CCommonThread::Start()
   30 {
   31          int ret = pthread_create(&m_nThreadId, NULL, ThreadEntry, (void*)this);
   32          if (ret != 0)
   33          {
   34                 WriteRunInfo::WriteLog("Create thread fail, error is %d.\n", strerror(errno));
   35                 return -1;
   36          }         
   37          return 0;
   38 }

   61 void *CCommonThread::ThreadEntry(void * arg)
   62 {
   63         CCommonThread *commonThread = (CCommonThread*)arg;
   64         int tmpRet = 0;
   65         tmpRet =  commonThread->Init();
   66         if (tmpRet != 0)
   67         {
   68                 //写日志， 表示线程启动失败
   69                 WriteRunInfo::WriteLog("The thread init fail, exit program");
   70                 exit(0);
   71         }
   72         commonThread->SetPid(getpid());
   73         commonThread->Run();
   74         return NULL;
   75 }
