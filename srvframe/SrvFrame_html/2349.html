20     void CStatItem::AddCommandStat(int nType, int nUsec, int nNetUsec,  int nOk)
21     {
22            m_mutex.Lock();       
23            if (nType >= 0 && nType < MAXCOMMANDSTATNUM && nUsec > 0)
24            {
25                     if (m_pCommandStat[nType].nUsed == 0)
26                     {
27                             m_mutex.UnLock();
28                             return;
29                     }
30                     
31                     
32                     if (nOk == 0)
33                     {
34                          m_pCommandStat[nType].nTotalOk ++;                     
35                     }
36                     else
37                     {
38                          m_pCommandStat[nType].nTotalFail ++;
39                     }
40                     if (nUsec > 0)  
41                             m_pCommandStat[nType].lTotalProcessTime += nUsec;
42                     if (nNetUsec > 0)
43                             m_pCommandStat[nType].lTotalNetProcessTime += nNetUsec;
44                     for (int i=0; i < PROCESSTIMENUM; i++)
45                     {
46                             if (i < PROCESSTIMENUM-1)
47                             {
48                                     if (nUsec > PROCESSTIME[i] && nUsec <= PROCESSTIME[i+1])
49                                     {
50                                             if (nOk == 0)
51                                             {
52                                                     m_pCommandStat[nType].nOk[i] ++;
53                                             }
54                                             else
55                                             {
56                                                     m_pCommandStat[nType].nFail[i] ++;
57                                             }
58                                             m_pCommandStat[nType].lProcessTime[i] += nUsec;
59                                             break;
60                                     }
61                             }
62                             else
63                             {
64                                     if (nOk == 0)
65                                     {
66                                             m_pCommandStat[nType].nOk[i] ++;
67                                     }
68                                     else
69                                     {
70                                             m_pCommandStat[nType].nFail[i] ++;
71                                     }
72                                     m_pCommandStat[nType].lProcessTime[i] += nUsec;
73                             }
74                     }                
75            }
76            m_mutex.UnLock();
77     }

95     void CStatItem::Update()
96     {      
97             int nowTime = CMyserver::GetInstance()->GetTimeVal().tv_sec;
98             if (nowTime - m_nPreUpdateTime < m_nIntervalTime)
99             {
100                     return;
101             }
102             PrintStat();
103             m_nPreUpdateTime = nowTime;                
104             int i=0;
105             for (i=0; i < MAXCOMMANDSTATNUM; i++)
106             {
107                     m_pCommandStat[i].nTotalOk = 0;
108                     m_pCommandStat[i].nTotalFail = 0;
109                     m_pCommandStat[i].lTotalProcessTime= 0;
110                     m_pCommandStat[i].lTotalNetProcessTime = 0;
111                     for (int j=0; j < PROCESSTIMENUM; j++)
112                     {
113                             m_pCommandStat[i].nOk[j] = 0;
114                             m_pCommandStat[i].nFail[j] = 0;
115                             m_pCommandStat[i].lProcessTime[j] = 0;
116                     }
117             }
118             
119             for (i=0; i < MAXCOMMANDSTATNUM; i++)
120             {
121                     m_pNormalStat[i].nType = 0;
122                     m_pNormalStat[i].nNum= 0;
123             }
124     
125     }

128     {        
129             //char *pEnd = m_statStr + snprintf(m_statStr, sizeof(m_statStr), "%s\n", m_sStatInfo.c_str());       
130             m_statStr.str("");

132             int i=0;                
133             for (i=0; i < MAXCOMMANDSTATNUM; i++)
134             {                
135                     if (m_pCommandStat[i].nUsed == 0)
136                     {
137                             continue;
138                     }
139                     
140                     int nAvgTime = 0;
141                     int nAvgNetTime = 0;
142                     int nTotalNum = m_pCommandStat[i].nTotalOk + m_pCommandStat[i].nTotalFail;
143                     if (nTotalNum > 0)
144                     {
145                             nAvgTime = m_pCommandStat[i].lTotalProcessTime/nTotalNum;                        
146                             nAvgNetTime  = m_pCommandStat[i].lTotalNetProcessTime/nTotalNum;                        
147                     }
148     //               pEnd += snprintf(pEnd, sizeof(m_statStr) - (pEnd-m_statStr), "[%s, %d, %d, %d, %d]", m_pCommandStat[i].sCmdName, m_pCommandStat[i].nTotalOk,

150                     m_statStr << "[" << m_pCommandStat[i].sCmdName << "," << m_pCommandStat[i].nTotalOk << "," 
151                         << m_pCommandStat[i].nTotalFail << "," << nAvgTime << "," << nAvgNetTime << "]";
152             }
153     
154             for (i=0; i < MAXCOMMANDSTATNUM; i++)
155             {
156                     if (m_pNormalStat[i].nUsed == 0)
157                     {
158                             continue;
159                     }
160                     //pEnd += snprintf(pEnd, sizeof(m_statStr) - (pEnd-m_statStr), "[%s:%d]", m_pNormalStat[i].sStatInfo, m_pNormalStat[i].nNum);
161                     m_statStr << "[" << m_pNormalStat[i].sStatInfo << ":" << m_pNormalStat[i].nNum << "]";
162             }
163             WriteRunInfo::WriteStat("%s", m_statStr.str().c_str());
164     }
