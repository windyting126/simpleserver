51     int CSendTask::HandleProcess(int events)
52     {        
53     //        DelFromEpoll();        
54             if ((events&MYPOLLERR) == MYPOLLERR || (events&MYPOLLHUP) == MYPOLLHUP)
55             {
56                     CMyserver::GetInstance()->GetCommonStat().AddNormalStat(SOCKETERRORTYPE);
57                     WriteRunInfo::WriteLog("Receive abnormity event %d", events);
58                     DestroySelf();
59                     return 0;
60             }  
61             int ret = 0;
62             //每次最多写10K
63             if (events &(MYPOLLOUT|MYPOLLRDNORM)) //可写事件
64             {               
65                     ret = SendData();               
66             }
67             else //其他事件
68             {
69                     DestroySelf();
70                     return 0;
71             }
72             
73             if (ret == -1 || ret == 1) //客户端已经关闭, 或者是句柄错误, 或者已经发送完毕
74             {
75                     DestroySelf();
76                     return 0;
77             }
78             //当这个任务还没有完的时候才能够加进去
79     //        AddToEpoll();
80             return 0;
81     }

108     int CSendTask::SendData()
109     {
110             if (m_pSendMsg == NULL)
111             {
112                     m_pSendMsg = GetSendMsg();
113                     if (m_pSendMsg == NULL)
114                     {
115                             WRITE_TRACE("Get send msg , but no msg");
116                             return 0;
117                     }
118                     m_iCurSendLen = 0;
119             }
120     
121             if (m_pSendMsg->GetBody() == NULL)
122             {
123                     WRITE_TRACE("The sendmsg body is NULL, exit the sendtask .....");
124                     return 1;
125             }
126      
127             
128             int retLen = 0;
129             int retValue = 0;
130             int sendLen = (SENDMAXLEN >  (m_pSendMsg->GetLen()- m_iCurSendLen))?
131                      (m_pSendMsg->GetLen()- m_iCurSendLen):SENDMAXLEN;
132             retLen = send( m_pTcpSocket->GetSocket(), m_pSendMsg->GetBody()+ m_iCurSendLen, 
133                     sendLen, MSG_DONTWAIT);
134             if (retLen == sendLen) //全部发送出去
135             {
136                     delete m_pSendMsg; //发送完毕,把消息删除掉
137                     m_pSendMsg = NULL;
138                     m_iCurSendLen = 0;
139                     retValue = 0;         
140                     //再取一个消息出来
141                     m_pSendMsg = GetSendMsg();
142                     if (m_pSendMsg != NULL)
143                     {
144                             if (m_pSendMsg->GetBody() == NULL)
145                             {
146                                     retValue = 1; //结束了
147                             }
148                     }                
149             }
150             else if (retLen > 0 && retLen < sendLen) //发送了一部分
151             {
152                     m_iCurSendLen += retLen;
153                     retValue = 0;
154             }
155             else //<= 0
156             {
157                     if (errno == ECONNRESET||
158                     errno == EDESTADDRREQ||
159                     errno == ENOTCONN||
160                     errno == EPIPE||
161                     errno == ETIMEDOUT || errno == EMSGSIZE)
162                     {
163                             retValue = -1;
164                     }
165                     //写日志
166                     WRITE_TRACE("Send data len invalid %d", retLen);                        
167             }
168             return retValue;
169     }

180     CSendMsg* CSendTask::GetSendMsg()
181     {
182             CSendMsg *pSendMsg = NULL;
183             if (m_pSendMsgList.size() == 0)
184             {
185                     return pSendMsg;
186             }
187             
188             list<CSendMsg*>::iterator listItr =  m_pSendMsgList.begin();
189             pSendMsg = *listItr;
190             //将这个项去掉
191             m_pSendMsgList.erase(listItr);
192             return pSendMsg;
193     }
