38     int CLongConnectedThread::Init( void )
39     {
40     	m_pEpollReactor = CEpollReactor::GetInstance();

42     	{
43     		return -1;

48     int  CLongConnectedThread::Process(CMsg *pMsg)
49     {
50             TMsg *pTMsg = pMsg->GetTMsg();
51     	if ( !pTMsg )
52     	{
53     		return 0;
54     	}
55             WRITE_TRACE("The LongConnected thread recv a  msg , msgtype %d...", pTMsg->msgType);        
56     	if ( pTMsg->msgType == CLIENTMSGTYPE)
57     	{
58     		CClientMsgBody *pBody = ( CClientMsgBody * )pMsg->GetMsgBody();
59     		if ( !pBody )
60     		{
61     			return 0;
62     		}
63     
64     		if ( m_iListType )
65     		{
66     			std::list<CSocketTask*>::iterator it = m_pSocketList.begin();
67     			while( it != m_pSocketList.end() )
68     			{
69     				CSocketTask * pSocketTask = ( *it );
70     				if ( pSocketTask )
71     				{
72     					pSocketTask->PutMsgToSendList( pBody->GetBuffer(), pBody->GetLen() );
73     				}
74     				it ++;
75     			}
76     		}
77     		else
78     		{
79     			map<ULONG64, CSocketTask*>::iterator it = m_pMapSocketList.begin();
80     			while( it != m_pMapSocketList.end() )
81     			{
82     				CSocketTask * pSocketTask = ( *it ).second;
83     				if ( pSocketTask )
84     				{
85     					pSocketTask->PutMsgToSendList( pBody->GetBuffer(), pBody->GetLen() );
86     				}
87     				it ++;
88     			}
89     		}
90     	}
91     	else if ( pTMsg->srcEntityType == EPOLLREACTORENTITY )
92     	{
93     		if (pTMsg->msgType == CONNECTMSGTYPE)
94     		{
95     			CConnectMsgBody *pBody = (CConnectMsgBody*)pMsg->GetMsgBody();
96     			if ( !pBody )
97     			{
98     				return 0;
99     			}
100     
101     			std::string tmpStr = pBody->GetRemoteIp();
102     			CTcpSocket *pSocket = new CTcpSocket( tmpStr, pBody->GetRemotePort(), pBody->GetSocket() );
103     			CSocketTask *tmpSocketTask = NULL;
104     			CCommonTaskArg commonTaskArg( pBody->GetDestMachineType(), pBody->GetDestMachineIndex(), true, m_iListType );
105     			tmpSocketTask = CMyserver::GetInstance()->GetClassFactory()->GenSocketTask(this, pSocket, COMMONTASK, &commonTaskArg );
106     			tmpSocketTask->Init();
107     			WriteRunInfo::WriteLog( "connect to %d %d %d succ ", ( int )pBody->GetDestMachineType(), ( int )pBody->GetDestMachineIndex(), pBody->GetSocket() );
108     		}
109     		else
110     		{
111     			WriteRunInfo::WriteLog("Send&recv  thread recv invalid msgtype %d", pTMsg->msgType);
112     		}
113     	}
114     	else if ( pTMsg->srcEntityType== MAINENTITY )
115     	{
116     		if (pTMsg->msgType == TOCONNECTMSGTYPE)
117     		{
118     			CToConnectMsgBody *pBody = (CToConnectMsgBody*)pMsg->GetMsgBody();
119     			if ( !pBody )
120     			{
121     				return 0;
122     			}
123     			if ( m_pEpollReactor )
124     			{
125     				m_pEpollReactor->Connect(pBody->GetDestMachineType(), pBody->GetDestMachineIndex(), pBody->GetRemoteIp(), pBody->GetRemotePort(), 3, m_nEntityType, m_nEntityId);
126     			}
127     		}
128     		else
129     		{
130     			WriteRunInfo::WriteLog("Send msg thread recv invalid msgtype %d", pTMsg->msgType);
131     		}
132     	}
133     	else if (pTMsg->msgType == DISCONNECTMSGTYP)
134     	{
135     		CConnectMsgBody *pBody = ( CConnectMsgBody* )pMsg->GetMsgBody();
136     		if ( !pBody )
137     		{
138     			return 0;
139     		}
140     		ULONG64 nTaskId = 0;
141     		if (CCommonTask::GetConnectTaskId(pBody->GetDestMachineType(), pBody->GetDestMachineIndex(), nTaskId) == 0)
142     		{
143     			if ( m_iListType )
144     			{
145     				std::list<CSocketTask*>::iterator it = m_pSocketList.begin();
146     				while( it != m_pSocketList.end() )
147     				{
148     					CSocketTask * pSocketTask = ( *it );
149     					if ( pSocketTask && pSocketTask->GetTaskId() == nTaskId)
150     					{
151     						delete pSocketTask;
152     						WriteRunInfo::WriteLog("LongConnectedThread : Destory task[%d,%d]", pBody->GetDestMachineType(), pBody->GetDestMachineIndex());
153     					}
154     					it ++;
155     				}
156     			}
157     			else
158     			{
159     				map<ULONG64, CSocketTask*>::iterator it = m_pMapSocketList.find(nTaskId);
160     				if( it != m_pMapSocketList.end() )
161     				{
162     					CSocketTask * pSocketTask = ( *it ).second;
163     					if ( pSocketTask )
164     					{
165     						delete pSocketTask;
166     						WriteRunInfo::WriteLog("LongConnectedThread : Destory task[%d,%d]", pBody->GetDestMachineType(), pBody->GetDestMachineIndex());
167     					}
168     				}
169     			}
170     		}
171     	}
172     	else
173     	{
174     		CNetProcessThread::Process( pMsg );
175     	}
176     	return 0;
177     }
