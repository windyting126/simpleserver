119     int CMyPoll::EpollWait()
120     {
121         int  nfds =  poll(m_pollfd, EPOLLSIZE, EPOLLWAITTIME);        
122         if (nfds <= 0)
123         {
124             if (nfds < 0)
125                 WriteRunInfo::WriteLog("poll nfds%d error(%d) is %s",nfds, errno, strerror(errno));      
126             return 0;
127         }
128         int tmpProcessNum=0;
129         for (int i =0; i < EPOLLSIZE; i++)
130         {
131             if ((m_pollfd[i].fd != -1) && (m_pollfd[i].events != 0) && (m_pollfd[i].revents != 0))
132             {
133                 m_taskList[i]->HandleProcess(m_pollfd[i].revents);
134                 tmpProcessNum++;
135                 if (m_taskList[i] != NULL) //如果句柄还存活的话把回调事件清零
136                 {
137                     m_pollfd[i].revents = 0;
138                 }
139             }
140             if (tmpProcessNum == nfds)
141             {
142                 break;
143             }
144         }
145         return nfds;
146     }
