   35 int CLongConnListenThread::Init()
   36 {
   37         int nRet = CNetProcessThread::Init();
   38         if (nRet != 0)
   39         {
   40                 return -1;
   41         }
   42 
   43         //生成一个监听句柄
   44         int nSocket;
   45         char szClientclientIp[INET_ADDRSTRLEN];
   46         int nIp = htonl(m_nHost);
   47         inet_ntop(AF_INET, &nIp, szClientclientIp, sizeof(szClientclientIp));
   48         nRet = CMySocket::CreateSock(SOCK_STREAM, szClientclientIp, m_nPort, nSocket);
   49         if (nRet < 0)
   50         {
   51                 WriteRunInfo::WriteLog("CLongConnListenThread::Init : Create listen sock fail %d", nRet);
   52                 return -1;
   53         }
   54         
   55         CTcpSocket* pTcpSocket = new CTcpSocket(szClientclientIp, m_nPort, nSocket);
   56         if (pTcpSocket == NULL)
   57         {
   58                 WriteRunInfo::WriteLog("CLongConnListenThread::Init : New listen socket object fail");
   59                 return -1;
   60         }
   61         
   62         //起监听端口                    
   63         CSocketTask* pListenTask = CMyserver::GetInstance()->GetClassFactory()->GenSocketTask(this, pTcpSocket, m_nListenTaskType);
   64         if (pListenTask == NULL)
   65         {
   66                 WriteRunInfo::WriteLog("CLongConnListenThread::Init : Gen Listen Task fail");
   67                 delete pTcpSocket;
   68                 return -1;
   69         }                      
   70         nRet = pListenTask->Init();
   71         if (nRet != 0)
   72         {
   73                 WriteRunInfo::WriteLog("CLongConnListenThread::Init : Init Listen Task fail");
   74                 return -1;
   75         }
   76         return 0;
   77 }
