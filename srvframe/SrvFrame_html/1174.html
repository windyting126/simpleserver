30     int  CShortConnectedProcessThread::Process(CMsg *pMsg)
31     {
32     	TMsg *pTMsg = pMsg->GetTMsg();
33     	if ( !pTMsg )
34     	{
35     		return 0;
36     	}
37             WriteRunInfo::WriteTrace("The ShortConnected thread recv a  msg , msgtype %d...", pTMsg->msgType);
38     
39     	if ( pTMsg->srcEntityType == LONGCONNECTEDENTITY )
40     	{
41     		if (pTMsg->msgType == CLIENTMSGTYPE )
42     		{
43     			CClientMsgBody *pBody = ( CClientMsgBody * )pMsg->GetMsgBody();
44     			if ( !pBody )
45     			{
46     				return 0;
47     			}
48                 CSocketTask *pSocketTask = FindSocketTask(pTMsg->taskId);
49                 if (pSocketTask != NULL)
50                 {
51                     pSocketTask->PutMsgToSendList( pBody->GetBuffer(), pBody->GetLen());
52                     pBody->DisAttachBuf(); //把buffer传到sockettask中了
53                 }
54                 
55     //			std::list<CSocketTask*>::iterator it = m_pSocketList.begin();
56     //			while( it != m_pSocketList.end() )
57     //			{
58     //				CSocketTask * pSocketTask = ( *it );
59     //				if ( pSocketTask )
60     //				{
61     //					pSocketTask->PutMsgToSendList( pBody->GetBuffer(), pBody->GetLen() );
62     //				}
63     //				it ++;
64     //			}
65     		}
66     		else
67     		{
68     			WriteRunInfo::WriteLog("Send&recv  thread recv invalid msgtype %d", pTMsg->msgType);
69     		}
70     	}
71     	else if ( pTMsg->srcEntityType == LISTENENTITY )
72     	{
73     		 if (pTMsg->msgType == RECVMSGTYPE)
74                     {
75                             CRecvMsgBody *pBody = (CRecvMsgBody*)pMsg->GetMsgBody();
76                             CTcpSocket *pSocket = new CTcpSocket(pBody->GetSockAddr(), sizeof(sockaddr_in), pBody->GetSocket());
77                             CSocketTask *tmpSocketTask = NULL;
78                             tmpSocketTask = CMyserver::GetInstance()->GetClassFactory()->GenSocketTask(this,  
79                                     pSocket, RECVANDSENDTASK);               
80                             tmpSocketTask->Init();         
81                             tmpSocketTask->SetTimeStat(pBody->GetTimeStat());
82                     }
83                     else
84                     {
85                             WriteRunInfo::WriteLog("Send&recv  thread recv invalid msgtype %d", pTMsg->msgType);
86                     }
87     	}
88     	else
89     	{
90     		CNetProcessThread::Process( pMsg );
91     	}
92     	return 0;
93     }
