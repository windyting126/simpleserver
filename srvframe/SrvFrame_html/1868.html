53     int CTimer::TimerCheck()
54     {
55             int tmpProcessNum = 0;
56             map<ULONG64, TTimerStr>::iterator mapItr;
57     		
58     		if ( m_nCurTimePoint >= m_nPreTimePoint )
59     		{
60     			for ( int iTimePoint = m_nPreTimePoint ; iTimePoint <= m_nCurTimePoint ; iTimePoint ++ )
61     			{
62     				mapItr = m_timerList[iTimePoint].begin();
63     				for (; mapItr != m_timerList[iTimePoint].end(); )
64     				{
65     					map<ULONG64, TTimerStr>::iterator tmpItr = mapItr;
66     					mapItr++;
67     					//发送超时消息
68     					SendTimerOutMsg(tmpItr->second);
69     					//删除这个节点
70     					m_timerList[iTimePoint].erase(tmpItr);
71     					tmpProcessNum ++;
72     				}
73     			}			
74     			m_nPreTimePoint = m_nCurTimePoint;
75     		}
76     		else
77     		{
78     			for ( int iTimePoint = m_nPreTimePoint ; iTimePoint <= TIMERCYCLE - 1 ; iTimePoint ++ )
79     			{
80     				mapItr = m_timerList[iTimePoint].begin();
81     				for (; mapItr != m_timerList[iTimePoint].end(); )
82     				{
83     					map<ULONG64, TTimerStr>::iterator tmpItr = mapItr;
84     					mapItr++;
85     					//发送超时消息
86     					SendTimerOutMsg(tmpItr->second);
87     					//删除这个节点
88     					m_timerList[iTimePoint].erase(tmpItr);
89     					tmpProcessNum ++;
90     				}
91     			}
92     			for ( int iTimePoint = 0 ; iTimePoint <= m_nCurTimePoint ; iTimePoint ++ )
93     			{
94     				mapItr = m_timerList[iTimePoint].begin();
95     				for (; mapItr != m_timerList[iTimePoint].end(); )
96     				{
97     					map<ULONG64, TTimerStr>::iterator tmpItr = mapItr;
98     					mapItr++;
99     					//发送超时消息
100     					SendTimerOutMsg(tmpItr->second);
101     					//删除这个节点
102     					m_timerList[iTimePoint].erase(tmpItr);
103     					tmpProcessNum ++;
104     				}
105     			}
106     			m_nPreTimePoint = m_nCurTimePoint;
107     		}
108             return tmpProcessNum;
109     }

111     int CTimer::SendTimerOutMsg(TTimerStr &timerStr)
112     {
113             TMsg tMsg;
114             tMsg.srcEntityId = 0;
115             tMsg.srcEntityType = TIMERENTITY;
116             tMsg.destEntityId = timerStr.iEntityId;
117             tMsg.destEntityType = timerStr.iEntityType;
118             tMsg.msgType = TIMERREACHTYPE;
119             tMsg.taskId = timerStr.iTaskId;
120             char *tmpBuf = new char[sizeof(TTimeEvent)];
121             memcpy(tmpBuf, &(timerStr.event), sizeof(TTimeEvent));
122             CBufMsgBody *pBufBody = new CBufMsgBody(tmpBuf, sizeof(TTimeEvent));
123             CMsg *pMsg = new CMsg(tMsg, pBufBody);
124     
125             CMyserver::GetInstance()->PutMsgToThread(pMsg);
126             return 0;
127     }
