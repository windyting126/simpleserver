   20 CTask::CTask(ULONG64  taskId, CWorkThread * pThread)
   21 { 
   22         m_iTaskId = taskId;   
   23         m_pThread = pThread;
   24         m_iTaskType = 0;
   25         m_status = INITSTATUS;
   26         m_pThread->AddTask(this);
   27         UpdateTaskNum(1);
   28         m_nStartTime = CMyserver::GetInstance()->GetTimeVal().tv_sec;
   29         m_nTaskTimeOut = -1;
   30 }
   31 
   32 CTask::~CTask()
   33 {
   34         WRITE_TRACE("Delete a  task %llu", m_iTaskId);
   35         //将任务从线程任务列表中删除
   36         m_pThread->DelTask(m_iTaskId);     
   37         //删除所有定时器             
   38         DeleteAllTimer();
   39         UpdateTaskNum(-1);
   40 }

   55 int CTask::CheckTimeOut()
   56 {        
   57         if (m_nTaskTimeOut == -1)
   58         {
   59                 return 0;
   60         }
   61         
   62         int ret = 0;
   63         if ((CMyserver::GetInstance()->GetTimeVal().tv_sec - m_nStartTime) > m_nTaskTimeOut)
   64         {
   65                 DestroySelf();
   66                 ret = 1;
   67         }
   68         return ret;
   69 }

   71 int CTask::DeleteAllTimer()
   72 {
   73         map<ULONG64, TTimeEvent>::iterator mapItr=m_timerList.begin();
   74         for (; mapItr != m_timerList.end(); )
   75         {
   76                 map<ULONG64, TTimeEvent>::iterator tmpItr = mapItr;
   77                 mapItr++;                           
   78                 SendTimerOutMsg(TIMERDELTYPE, tmpItr->second);
   79                 m_timerList.erase(tmpItr);        
   80         }
   81         return 0;
   82 }

   84 int CTask::AddOneTimer(int event, int timeLength)
   85 {
   86         TTimeEvent tmpEvent;
   87 
   88         ULONG64 iTimerId = CTimer::GentTimerId();
   89 
   90         tmpEvent.m_iTimerId = iTimerId;
   91         tmpEvent.m_nEvent = event;
   92         tmpEvent.m_nTimerPoint = (CMyserver::GetInstance()->GetTimerPoint() + timeLength)%TIMERCYCLE;        
   93         
   94         m_timerList.insert(make_pair(iTimerId, tmpEvent));
   95 
   96         //需要再考虑下
   97         SendTimerOutMsg(TIMERADDTYPE, tmpEvent);
   98         return 0;
   99 }

  102 int CTask::DeleteOneTimer(ULONG64 iTimerId)
  103 {       
  104         //删除自身的定时器
  105         map<ULONG64, TTimeEvent>::iterator mapItr = m_timerList.find(iTimerId);
  106         if (mapItr != m_timerList.end())
  107         {
  108                  //删除定时器    
  109                 SendTimerOutMsg(TIMERDELTYPE, mapItr->second);
  110                 m_timerList.erase(mapItr);               
  111         }
  112         return 0;
  113 }

  128 int CTask::SendTimerOutMsg(int msgType,  TTimeEvent &timerEvent)
  129 {
  130         TMsg tMsg;
  131         tMsg.srcEntityId = m_pThread->GetEntityId();
  132         tMsg.srcEntityType = m_pThread->GetEntityType();
  133         tMsg.destEntityId = 0;
  134         tMsg.destEntityType = TIMERENTITY;
  135         tMsg.msgType = msgType;
  136         tMsg.taskId = m_iTaskId; 
  137         char *tmpBuf = new char[sizeof(TTimeEvent)];
  138         memcpy(tmpBuf, &timerEvent, sizeof(TTimeEvent));        
  139         CBufMsgBody *pBufBody=  new CBufMsgBody(tmpBuf, sizeof(TTimeEvent));        
  140         CMsg *pMsg = new CMsg(tMsg, pBufBody);        
  141         CMyserver::GetInstance()->PutMsgToThread(pMsg);
  142         return 0;
  143 }

  145 ULONG64 CTask::GentTaskId()
  146 {        
  147         ULONG64        ret = 0;
  148         m_taskIdMutex.Lock();
  149         m_lCurTaskId++;
  150         ret = m_lCurTaskId;
  151         m_taskIdMutex.UnLock();
  152         return ret;
  153 }
