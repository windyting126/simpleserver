38     int CLongConnRecvAndSendThread::Process(CMsg* pMsg)
39     {
40     	TMsg* pTMsg = pMsg->GetTMsg();
41     
42     	if (pTMsg->msgType == MSGTYPE_LONGCONN_COMMON_RES)
43     	{
44     		CLongConnBufMsgBody* pMsgBody = (CLongConnBufMsgBody*)pMsg->GetMsgBody();
45     		if (!pMsgBody)
46     		{
47     			return 0;
48     		}	
49     		CSocketTask* pSocketTask = FindSocketTask(pTMsg->taskId);
50     		if (pSocketTask)
51     		{
52     			pSocketTask->PutMsgToSendList(pMsgBody->GetBuf(), pMsgBody->GetBufLen());
53     			pMsgBody->DetachBuf();
54     		}
55     	}
56     	else if (pTMsg->msgType == MSGTYPE_LONGCONN_ACCEPT)
57     	{
58     		CRecvMsgBody* pBody = (CRecvMsgBody*)pMsg->GetMsgBody();
59     		CTcpSocket* pSocket = new CTcpSocket(pBody->GetSockAddr(), sizeof(sockaddr_in), pBody->GetSocket());
60     		CSocketTask* pTmpSocketTask = NULL;
61     		pTmpSocketTask = CMyserver::GetInstance()->GetClassFactory()->GenSocketTask(this, pSocket, LONGCONNRECVANDSENDTASK);
62     		pTmpSocketTask->Init();
63     		pTmpSocketTask->SetTimeStat(pBody->GetTimeStat());
64     	}
65     	return 0;
66     }
