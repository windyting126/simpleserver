   92 int CLongConnRecvAndSendTask::HandleProcess(int nEvent)
   93 {
   94         if ((nEvent & MYPOLLERR) == MYPOLLERR || (nEvent & MYPOLLHUP) == MYPOLLHUP)
   95         {
   96                 WriteRunInfo::WriteLog("CLongConnRecvAndSendTask::HandleProcess: socket error, event=%d socket=%d error=%d [threadtype = %d]", nEvent, m_pTcpSocket->GetSocket(), errno, m_pThread->GetEntityType());
   97                 CMyserver::GetInstance()->GetCommonStat().AddNormalStat(SOCKETERRORTYPE);
   98                 DestroySelf();
   99                 return 0;
  100         }
  101 
  102         m_nStartTime = CMyserver::GetInstance()->GetTimeVal().tv_sec;
  103         
  104         int nRet = 0;
  105         if ((nEvent & MYPOLLIN) == MYPOLLIN)
  106         {
  107                 nRet = RecvProcess();
  108                 if (nRet == -1)
  109                 {
  110                         DestroySelf();
  111                         return -1;
  112                 }
  113         }
  114         
  115         if ((nEvent & MYPOLLOUT) == MYPOLLOUT)
  116         {
  117                 nRet = SendProcess(); 
  118                 if (nRet == -1)
  119                 {
  120                         DestroySelf();
  121                         return -1;                      
  122                 }
  123         }               
  124         return 0;
  125 }

  143 int CLongConnRecvAndSendTask::RecvProcess()
  144 {
  145         unsigned int uMaxReceive;
  146     int nRecvLen = 0;
  147 
  148         int ii = 0;
  149         while (ii < LONGCONNECT_RECVSNED_NUM)
  150         {
  151                 uMaxReceive = m_uReceiveSize - m_uReceivePos;
  152                 nRecvLen = recv(GetSocket(), &m_vReceiveBuffer[0] + m_uReceivePos,  uMaxReceive, MSG_DONTWAIT);
  153 
  154                 if (nRecvLen == 0)
  155                 {
  156                         WriteRunInfo::WriteTrace("CLongConnRecvAndSendTask::RecvProcess : recv client close , close the socket");
  157                         return -1;
  158                 }
  159                 
  160                 if (nRecvLen < 0)
  161                 {
  162                         if (errno != EINTR && errno != EAGAIN)
  163                         {
  164                                 WriteRunInfo::WriteLog("CLongConnRecvAndSendTask::RecvProcess : recv client error(%d) , close the socket", errno);
  165                                 return -1;
  166                         }
  167                         return 0;
  168                 }
  169                 
  170                 m_uReceivePos += nRecvLen;
  171                 switch(m_state) 
  172                 {
  173                         case ReadPacketHead:
  174                                 {
  175                                         if (!PacketFinished()) break;
  176                                         int iNeedRecvLen =  TFrameLongConnectHead::GetBodyLen((char*)(&m_vReceiveBuffer[0]), m_uReceivePos);
  177                                         if (iNeedRecvLen > LONGCONNECT_MAX_BUFFLEN || iNeedRecvLen < 0)
  178                                         {
  179                                                 return -1;
  180                                         }
  181                                         if (iNeedRecvLen == 0)  //如果长连接头后面长度为0，表明只是ping包！
  182                                         {
  183                                                 char* pBuf = new char[m_uReceivePos];
  184                                                 memcpy(pBuf, (char*)(&m_vReceiveBuffer[0]), m_uReceivePos);
  185                                                 PutMsgToSendList(pBuf, m_uReceivePos);
  186                                                 ResetRecvBuffer(0, TFrameLongConnectHead::FLCH_HEADLEN);        
  187                                         }
  188                                         else
  189                                         {
  190                                                 m_state = ReadPacket;
  191                                                 ResetRecvBuffer(TFrameLongConnectHead::FLCH_HEADLEN, iNeedRecvLen + TFrameLongConnectHead::FLCH_HEADLEN);
  192                                         }                       
  193                                         break;
  194                                 }
  195                         case ReadPacket:
  196                                 {
  197                                         if (!PacketFinished()) break;
  198                                         if (ProcessPacket() < 0)
  199                                         {
  200                                                 return -1;
  201                                         }
  202 
  203                                         m_state = ReadPacketHead;
  204                                         ResetRecvBuffer(0, TFrameLongConnectHead::FLCH_HEADLEN);
  205                                         break;
  206                                 }
  207                         default:
  208                                 {
  209                                         WriteRunInfo::WriteError("CLongConnRecvAndSendTask::RecvProcess : wrong state!!");
  210                                         return -1;
  211                                         break;
  212                                 }
  213                 }
  214                 ii++;
  215         }
  216         return 0;
  217 }

  219 int CLongConnRecvAndSendTask::SendProcess()
  220 {
  221         int nRet = 0;
  222         if (m_pCurSendMsg != NULL)
  223         {
  224                 nRet = SendData();
  225         }
  226         else
  227         {
  228                 m_pCurSendMsg = GetMsgFromSendList();
  229                 if (m_pCurSendMsg != NULL)
  230                 {
  231                         nRet = SendData(); 
  232                 } 
  233                 else
  234                 {
  235                         ModifyEpollDelWrite();
  236                         nRet = 1;
  237                 }
  238         }
  239         return nRet;
  240 }

  242 int CLongConnRecvAndSendTask::ProcessPacket()
  243 {
  244         char* pBuf = (char*)(&m_vReceiveBuffer[0]);
  245         int nBufLen = (int)m_uReceiveSize;
  246         
  247         CBufMgr BufMgr(pBuf, nBufLen);
  248         TFrameLongConnectHead FrameLongConnectHead;
  249         int nRet = FrameLongConnectHead.Decode(BufMgr);
  250         BufMgr.DistachBuf();
  251         if (nRet != 0)
  252         {
  253                 return -1;
  254         }
  255         
  256         int iLength = FrameLongConnectHead.m_nBodyLen;
  257         if(iLength + TFrameLongConnectHead::FLCH_HEADLEN != nBufLen)
  258         {
  259                 return -1;
  260         }
  261 
  262         //redifine function ProcessLongConnMsg
  263         ProcessLongConnMsg(pBuf + TFrameLongConnectHead::FLCH_HEADLEN, iLength, FrameLongConnectHead);
  264 
  265         return nRet;
  266 }
