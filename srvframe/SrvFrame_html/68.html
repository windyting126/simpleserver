   23 int CFileReqHttpCmd::Process()
   24 {      
   25         try
   26         {
   27                 if (m_sUrl.find("..") != string::npos)
   28                 {
   29                         m_nResult = -1;
   30                         return -1;
   31                 }               
   32                 
   33                                 
   34                 string sFile = m_sFileRootDir + m_sUrl;
   35                 CMyFile myfile(sFile);
   36                 myfile.FOpen("r");
   37                 struct stat *pStat = myfile.GetStatBuf();
   38                                 m_sFileModTime = CToolKit::SecToHttpTime(pStat->st_mtime);
   39                                 string sOut;
   40                             int ret = CToolKit::GetStringByLabel(m_sInHead, "If-Modified-Since:", "\r\n", sOut);
   41                                 if (ret == 0) //有If-Modified-Since:
   42                                 {
   43                                         time_t modTime = CToolKit::TimeToSec(sOut);
   44                                         if (pStat->st_mtime == modTime)
   45                                         {
   46                                                 m_nResult = 1;
   47                                                 return 0;
   48                                         }
   49                                 }
   50                 char *pBuf = new char[pStat->st_size];
   51                 CBufMgr bufMgr(pBuf); //自动释放
   52                 
   53                 myfile.FRead(pBuf, pStat->st_size);
   54                 m_sOutBody = string(pBuf, pStat->st_size);
   55         }
   56         catch(CMyException &e)
   57         {
   58                 m_nResult = -1;
   59                 WriteRunInfo::WriteError("Error, %s", e.GetErrMsg().c_str());
   60         }
   61         return 0;
   62 }

   69 int CFileReqHttpCmd::Encode()
   70 {
   71         if (m_nResult < 0)
   72         {
   73             m_sResHeader =  "HTTP/1.1 404 NOT FOUND\r\n";
   74         }
   75                 else if (m_nResult == 1) //304
   76                 {
   77                         m_sResHeader = "HTTP/1.1 304 Not Modified\r\n";
   78                         m_sResHeader += "Last-Modified: "+ m_sFileModTime + "\r\n"; 
   79                 }
   80         else //==0
   81         {
   82                 m_sResHeader += "Last-Modified:"+ m_sFileModTime + "\r\n";
   83             string sSuffix;
   84             CToolKit::GetStringByLabel(m_sUrl, ".", "", sSuffix);
   85             map<string, string>::iterator mapItr = m_mineType.m_minetype.find(sSuffix);
   86             if (mapItr != m_mineType.m_minetype.end())
   87             {
   88                 m_sResHeader += "Content-Type: " + mapItr->second + "\r\n";
   89             }
   90         }
   91         return 0;
   92 }
