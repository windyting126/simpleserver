  288 template<typename T>
  289 int CHttpRecvAndSendTask<T>::SendToList(char *pBuf, int nLen)
  290 {       
  291         if (pBuf == NULL || nLen == 0)
  292         {       
  293                 return -1;
  294         }
  295         //由于设计缺陷，此处更改违反open-close
  296         //为不影响已上线的服务器，增加OLD_MODE/NEW_MODE
  297         string strHTTPHead;
  298         long long llResContentLen;
  299         llResContentLen = (OLD_MODE == m_nServerMode) ? nLen : m_llResContentLen;
  300         if (m_bEncodeHttpHead)
  301         {
  302                 char BodLen[12];
  303                 memset(BodLen, 0 , sizeof(BodLen));
  304                 snprintf(BodLen, sizeof(BodLen), "%lld", llResContentLen);
  305                 strHTTPHead = GEN_HTTP_RET_HEAD(strHTTPHead, BodLen, m_strResponseCode, m_strResponsePhrase);
  306         }
  307 
  308         char *pHttpReHeadBuf = new char[strHTTPHead.length() + nLen];
  309         memcpy(pHttpReHeadBuf, strHTTPHead.c_str(), strHTTPHead.length());
  310         memcpy(pHttpReHeadBuf + strHTTPHead.length(), pBuf, nLen);
  311         delete [] pBuf;
  312         CSendMsg *pSendMsg = new CSendMsg(strHTTPHead.length() + nLen, pHttpReHeadBuf);
  313         m_sendList.push_back(pSendMsg);
  314 
  315         m_llHaveResLen += nLen;  //已经送至发送队列的数据长度
  316 
  317         int tmpEvent = MYPOLLERR|MYPOLLHUP|MYPOLLIN| MYPOLLOUT;
  318         CNetProcessThread* pNetThread = (CNetProcessThread*)m_pThread;
  319         CEpoll* pEoll = pNetThread->GetEpoll();
  320         pEoll->ModifyFromEpoll(this, tmpEvent);
  321 
  322         return 0;
  323 }

  457 template<typename T>
  458 int CHttpRecvAndSendTask<T>::ParsePackets(char *pPacket, unsigned int uiPacketLen)
  459 {
  460         char *pCurPos = pPacket;
  461         char *pEndPos = pPacket + uiPacketLen;
  462         int nRet = 0;
  463         unsigned int nRestLen = pEndPos - pCurPos;
  464         while (1)
  465         {
  466                 char *pTmp = strstr(pCurPos, "\r\n\r\n");
  467                 if(pTmp == NULL)
  468                 {
  469                         m_state = ReadHttpHead;
  470                         break;
  471                 }
  472                 //Get Http Head
  473                 m_sHttpHead = string(pCurPos, pTmp - pCurPos + 4);
  474 
  475                 //Get Http Head Method
  476                 char* pMethodTmp = strstr(pCurPos, " ");
  477                 if (NULL == pMethodTmp)
  478                 {
  479                         WriteRunInfo::WriteLog("ParsePackets - There is not Http Head Method ");
  480                         WriteRunInfo::WriteLog("%s", m_sHttpHead.c_str());
  481                         return -1;
  482                 }       
  483                 string strMethod(pCurPos, pMethodTmp - pCurPos);
  484                 if (0 == strMethod.compare("GET"))
  485                 {
  486                         m_nContentLen = 0;
  487                         m_iBodyHaseRecvLen = 0;
  488                         m_state = ReadHttpHead;
  489                         nRet = ProcessGetMethod(m_sHttpHead.c_str(), m_sHttpHead.length());
  490                         if (nRet != 0)
  491                         {
  492                                 WriteRunInfo::WriteLog("ProcessGetMethod Failed");
  493                                 WriteRunInfo::WriteLog("Buf = %s", m_sHttpHead.c_str());
  494                                 return -1;
  495                         }
  496 
  497                         pCurPos += m_sHttpHead.length();
  498                         nRestLen = pEndPos - pCurPos;
  499                         m_sHttpHead.clear();
  500                 }
  501                 else if (0 == strMethod.compare("HEAD"))
  502                 {
  503                         m_nContentLen = 0;
  504                         m_iBodyHaseRecvLen = 0;
  505                         m_state = ReadHttpHead;
  506                         nRet = ProcessHeadMethod(m_sHttpHead.c_str(), (int)m_sHttpHead.length());
  507                         if (nRet != 0)
  508                         {
  509                                 WriteRunInfo::WriteLog("ProcessHeadMethod Failed");
  510                                 WriteRunInfo::WriteLog("Buf = %s", m_sHttpHead.c_str());
  511                                 return -1;
  512                         }
  513 
  514                         pCurPos += m_sHttpHead.length();
  515                         nRestLen = pEndPos - pCurPos;
  516                         m_sHttpHead.clear();
  517                 }
  518                 else if (0 == strMethod.compare("POST"))
  519                 {
  520                         string sOut;
  521                         int nTmp;               
  522                         //find content-len
  523                         nTmp = CToolKit::GetStringByLabel(m_sHttpHead, "Content-Length:", "\r\n", sOut, 0, 0);
  524                         if(nTmp != 0)
  525                         {
  526                                 nTmp = CToolKit::GetStringByLabel(m_sHttpHead, "Content-length:", "\r\n", sOut, 0, 0);
  527                                 if (nTmp != 0)
  528                                 {
  529                                         WriteRunInfo::WriteLog("There is no Content-Length");
  530                                         return -1;
  531                                 }
  532                         }
  533                         m_nContentLen = strtoull(sOut.c_str(), NULL, 10);
  534 
  535                         if (m_nContentLen > (pEndPos - pCurPos) - m_sHttpHead.length())
  536                         {
  537                                 m_state = ReadHttpBody;
  538                                 m_iBodyHaseRecvLen = (pEndPos - pCurPos) - m_sHttpHead.length();
  539                                 break;
  540                         }
  541                         nRet = ParseOnePacket(pTmp + 4, m_nContentLen);
  542                         if(nRet != 0)
  543                         {
  544                                 WRITE_TRACE("ParseOnePacket Failed");
  545                                 WRITE_TRACE("Buf = %s", pCurPos);
  546                                 return -1;
  547                         }
  548                         
  549                         pCurPos += (m_sHttpHead.length() + m_nContentLen);
  550                         nRestLen = pEndPos - pCurPos;
  551 
  552                         m_sHttpHead.clear();
  553                         m_iBodyHaseRecvLen = 0;
  554                         m_nContentLen = 0;
  555                 }
  556                 else
  557                 {
  558                         WRITE_TRACE("Do not support the method %s", strMethod.c_str());
  559                         WRITE_TRACE("%s", m_sHttpHead.c_str());
  560                         return -1;
  561                 }
  562         }
  563         
  564         nRestLen = pEndPos - pCurPos;
  565         m_iHasRecvLen = nRestLen;
  566         if (nRestLen > 0)
  567         {
  568                 if (nRestLen <= m_iRecvBufLen)
  569                 {
  570                         char TmpBuf[m_iRecvBufLen];
  571                         memcpy(TmpBuf, pCurPos, nRestLen);
  572                         memset(m_newRecvBuf, 0, m_iRecvBufLen);
  573                         memcpy(m_newRecvBuf, TmpBuf, nRestLen);
  574                 }
  575                 else
  576                 {
  577                         nRet = -1; 
  578                 }               
  579         }
  580         else
  581         {
  582                 ReSetRecvParameter();
  583         }
  584         return nRet;    
  585 }

  595 template<typename T>
  596 int CHttpRecvAndSendTask<T>::GetHttpHead()
  597 {
  598         int nRet=0;
  599 
  600         int needRecv = m_iHttpHeadLen - m_iHasRecvLen;
  601         if (needRecv <= 0) 
  602         {   
  603                 WriteRunInfo::WriteLog("The httphead is too large");
  604                 return -1;
  605         }
  606 
  607         int iReced = recv(m_pTcpSocket->GetSocket(), m_newRecvBuf + m_iHasRecvLen, needRecv, MSG_DONTWAIT);
  608 
  609         WriteRunInfo::WriteTrace("HEAD : Http Recv m_iHasRecvLen = %d", m_iHasRecvLen);
  610         WriteRunInfo::WriteTrace("HEAD : Http Recv Len = %d", iReced);
  611         nRet = -1;
  612         do 
  613         {
  614                 //Recv socket close=======
  615                 if (iReced == 0)
  616                 {
  617                         WriteRunInfo::WriteTrace("the httpclient close the socket %d", m_pTcpSocket->GetSocket());
  618                         break;
  619                 }
  620 
  621                 //Recv error==============
  622                 if(iReced < 0)
  623                 {
  624                         if ( errno != EINTR && errno != EAGAIN )
  625                         {
  626                                 WriteRunInfo::WriteLog("socket error , errnor is %d", errno);
  627                         }
