47     int CRecvAndSendTask::HandleProcess(int events)
48     {        
49             if ((events&MYPOLLERR) == MYPOLLERR || (events&MYPOLLHUP) == MYPOLLHUP)
50             {
51                     WriteRunInfo::WriteLog("Recv a invalid data pack, event %d, close the socket %d error %d [threadtype = %d]", events, 
52                             m_pTcpSocket->GetSocket(), errno, m_pThread->GetEntityType());
53                     CMyserver::GetInstance()->GetCommonStat().AddNormalStat(SOCKETERRORTYPE);
54                     DestroySelf();
55                     return 0;
56             }
57             //不断改变开始时间, 如果超过２分钟没有响应,就会自动断开 
58             m_nStartTime = CMyserver::GetInstance()->GetTimeVal().tv_sec;
59             
60             int ret = 0;
61             if ((events& MYPOLLIN) == MYPOLLIN) //读事件
62             {
63                     ret = RecvProcess();
64                     if (ret == -1)
65                     {
66                         DestroySelf();
67                         return -1;
68                     }
69             }
70             
71             if ((events& MYPOLLOUT) == MYPOLLOUT) //写事件
72             {
73                     ret =  SendProcess(); 
74                     if ( ret == -1 )
75                     {
76                         DestroySelf();
77                         return -1;                	
78                     }
79             }               
80             return 0;
81     }

83     int CRecvAndSendTask::AddToEpoll()
84     {          
85             int tmpEvent = MYPOLLERR|MYPOLLHUP|MYPOLLET|MYPOLLIN;                         
86             if (m_sendList.GetNum() != 0 || m_pCurSendMsg != NULL)
87             {
88                     tmpEvent = tmpEvent|MYPOLLOUT;
89             }
90             
91             CNetProcessThread *pNetThread = (CNetProcessThread*)m_pThread;
92             CEpoll *pEoll = pNetThread->GetEpoll();
93             pEoll->AddToEpoll(this, tmpEvent);
94             return 0;
95     }

98     CSendMsg *CRecvAndSendTask::GetMsgFromSendList()
99     {
100             if (m_sendList.begin() == m_sendList.end())
101             {
102                     return NULL;
103             }
104             list<CSendMsg*>::iterator itr = m_sendList.begin();
105             CSendMsg *pMsg = (*itr);
106             m_sendList.erase(itr);
107             return pMsg;                
108     }

111     int CRecvAndSendTask::PutMsgToSendList(const char *pBuf, int nLen)
112     {        
113             if (pBuf == NULL || nLen == 0)
114             {       
115                     return -1;
116             }               
117             char *pTmpBuf = new char[nLen];
118             memcpy(pTmpBuf, pBuf, nLen);
119             CSendMsg *pSendMsg = new CSendMsg(nLen, pTmpBuf);
120             
121             m_sendList.push_back(pSendMsg);
122             int tmpEvent = MYPOLLERR|MYPOLLHUP|MYPOLLET|MYPOLLIN|MYPOLLOUT;
123             CNetProcessThread *pNetThread = (CNetProcessThread*)m_pThread;
124             CEpoll *pEoll = pNetThread->GetEpoll();
125             pEoll->ModifyFromEpoll(this, tmpEvent);
126             return 0;
127     }

129     int CRecvAndSendTask::Process(CMsg *pMsg)
130     {
131             TMsg *pTMsg = pMsg->GetTMsg();
132             if (pTMsg->msgType == SENDMSGTYPE)
133             {
134                     CSendMsgBody *pBody = (CSendMsgBody*)pMsg->GetMsgBody();
135                     pBody->SetSendBuf(NULL);                
136                     PutMsgToSendList(pBody->GetSendBuf(), pBody->GetSendLen());
137             }
138             else
139             {
140                     WriteRunInfo::WriteLog("The task recv a invalid msg %d", pTMsg->msgType);
141             }
142             return 0;
143     }

146     int CRecvAndSendTask::SendData()
147     {         
148             int retLen = 0;
149             int retValue = 0;
150             unsigned int sendLen = (SENDMAXLEN >  (m_pCurSendMsg->GetLen()- m_iCurSendLen))?
151                      (m_pCurSendMsg->GetLen()- m_iCurSendLen):SENDMAXLEN;
152             retLen = send( m_pTcpSocket->GetSocket(), m_pCurSendMsg->GetBody()+ m_iCurSendLen, 
153                     sendLen, MSG_DONTWAIT);
154            // WRITE_TRACE("send data %d",retLen);
155             if (retLen > 0)
156             {               
157                     m_iCurSendLen += retLen;
158                     if (m_iCurSendLen == m_pCurSendMsg->GetLen()) //全部发送出去
159                     {                
160                             delete m_pCurSendMsg;
161                             m_pCurSendMsg = NULL;
162                             m_iCurSendLen = 0;
163                             retValue = 1;         
164                     }
165                     else  //发送了一部分
166                     {                        
167                             retValue = 0;
168                     }
169             }
170             else //<= 0
171             {
172                     if (errno == ECONNRESET||
173                     errno == EDESTADDRREQ||
174                     errno == ENOTCONN||
175                     errno == EPIPE||
176                     errno == ETIMEDOUT || errno == EMSGSIZE)
177                     {
178                             WriteRunInfo::WriteLog("Send error %d:%s", errno, strerror(errno));
179                             return -1;
180                     }
181                     else
182                     {
183                         //写日志
184                         WriteRunInfo::WriteTrace("Send data len invalid %d", retLen);
185                         return 2;
186                     }
187             }
188             return retValue;
189     }

192     int CRecvAndSendTask::SendBufProcess(const char * pBuf, int nLen)
193     {
194             if (pBuf == NULL || nLen == 0)
195             {
196                     return -1;
197             }
198             
199             int retValue = 0;
200             int ret = send(m_pTcpSocket->GetSocket(), pBuf, nLen, MSG_DONTWAIT);
201             if (ret == nLen) //发送完毕
202             {
203                     retValue =  -1;
204             }
205             else if (ret >= 0 && ret < nLen) //将剩下的放入队列
206             {
207     
208                     PutMsgToSendList(pBuf + ret, nLen - ret);
209                     retValue =  0;
210             }
211             else //<　０
212             {
213                     if (errno != EINTR && errno != EAGAIN)
214                     {
215                             WriteRunInfo::WriteLog("send fail, error(%d) is %s", errno, strerror(errno));
216                             CMyserver::GetInstance()->GetCommonStat().AddNormalStat(SOCKETERRORTYPE);
217                             retValue = -1;
218                     }          
219                     else    //继续接收消息
220     
221                     {
222                             WRITE_TRACE("EAGAIN, EINTR ");
223                             PutMsgToSendList(pBuf, nLen);
224                             retValue = 0;
225                     }                        
226             }        
227             return retValue;
228     }
