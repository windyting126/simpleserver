3     int CMultiListenThread::Init()
4     {
5     	int retValue = CListenThread::Init();
6     	if (retValue != 0)
7     	{
8     		return -1;
9     	}
10     	CThreadLock ThreadLock(&m_ListenListLock);
11     	list<TListenInfo>::iterator itrListen = m_lstListenInfo.begin();
12     	while (itrListen != m_lstListenInfo.end())
13     	{
14     		string sHost = itrListen->m_strHost;
15     		unsigned short nPort = itrListen->m_nPort;
16     		itrListen->ListenBlock.mutex.Lock();
17     		if (itrListen->ListenBlock.nSock == -1)
18     		{
19     			//生成一个监听句柄
20     			int nSocket;                           
21     			retValue = CMySocket::CreateSock(SOCK_STREAM, sHost, nPort, nSocket);
22     			if (retValue < 0)
23     			{
24     				WriteRunInfo::WriteLog("Create listen sock fail %d", retValue);
25     				itrListen->ListenBlock.mutex.UnLock();
26     				return -1;
27     			}
28     			itrListen->ListenBlock.nSock=nSocket;
29     		}
30     		itrListen->ListenBlock.mutex.UnLock();
31     
32     		CTcpSocket *pTcpSocket = new CTcpSocket(sHost, nPort, itrListen->ListenBlock.nSock);
33     		if (pTcpSocket == NULL)
34     		{
35     			WriteRunInfo::WriteLog("New listen socket object fail");
36     			return -1;
37     		}
38     
39     		//起监听端口                    
40     		CListenTask* pListenTask = (CListenTask*)CMyserver::GetInstance()->GetClassFactory()->GenSocketTask(this, pTcpSocket,  LISTENTASK);        
41     		
42     		if (pListenTask == NULL)
43     		{
44     			WriteRunInfo::WriteLog("Malloc memory fail or listen fail");
45     			return -1;
46     		}
47     		retValue = pListenTask->Init();
48     		if (retValue != 0)
49     		{
50     			WriteRunInfo::WriteLog("Init Listen Task Fail(%s:%d)", sHost.c_str(), nPort);
51     			return -1;
52     		}
53     		itrListen ++;
54     	}
55     	return 0;
56     }
